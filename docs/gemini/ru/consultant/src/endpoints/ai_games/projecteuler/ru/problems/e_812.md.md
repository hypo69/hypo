"""
Модуль для решения задачи 812 Project Euler.
=========================================================================================

Этот модуль находит сумму первых k простых чисел в последовательности Фибоначчи.

Пример использования
--------------------

.. code-block:: python

    result = solve_problem_812(100)
    print(f"Сумма первых 100 простых чисел Фибоначчи: {result}")
"""
from typing import List
import math  # Импорт модуля math для математических операций
from src.logger.logger import logger  # Импорт логера

def is_prime(n: int) -> bool:
    """
    Проверяет, является ли число простым.

    :param n: Число для проверки.
    :return: True, если число простое, иначе False.
    """
    if n <= 1:
        return False
    if n <= 3:
        return True
    if n % 2 == 0 or n % 3 == 0:
        return False
    i = 5
    while i * i <= n:
        if n % i == 0 or n % (i + 2) == 0:
            return False
        i += 6
    return True

def fibonacci_generator() -> int:
    """
    Генерирует числа Фибоначчи.

    :yield: Следующее число Фибоначчи.
    """
    a, b = 0, 1
    while True:
        yield a
        a, b = b, a + b

def find_sum_of_first_k_fibonacci_primes(k: int) -> int:
    """
    Находит сумму первых k простых чисел в последовательности Фибоначчи.

    :param k: Количество простых чисел Фибоначчи для суммирования.
    :return: Сумма первых k простых чисел Фибоначчи.
    """
    primes_found = 0
    total_sum = 0
    fib_gen = fibonacci_generator()
    try:
        while primes_found < k:
            fib_num = next(fib_gen)
            if fib_num > 1 and is_prime(fib_num):
                total_sum += fib_num
                primes_found += 1
    except Exception as e:
        logger.error(f"Произошла ошибка при поиске простых чисел Фибоначчи: {e}")
        return 0 # Возвращаем 0 в случае ошибки
    return total_sum

def solve_problem_812(k: int) -> int:
    """
    Решает задачу 812 Project Euler, вычисляя сумму первых k простых чисел Фибоначчи.

    :param k: Количество простых чисел Фибоначчи для суммирования.
    :return: Сумма первых k простых чисел Фибоначчи.
    """
    result = find_sum_of_first_k_fibonacci_primes(k)
    return result

if __name__ == '__main__':
    # Код исполняет решение задачи для k=100
    result = solve_problem_812(100)
    print(f"Сумма первых 100 простых чисел Фибоначчи: {result}")
