"""
Модуль для решения задачи 357 Project Euler: Простое порождение делителей.
===========================================================================

Модуль содержит функции для определения простых чисел и проверки, является ли число
простым порождающим делители числом. Основная функция вычисляет сумму всех простых
порождающих делителей чисел до заданного предела.

Пример использования
--------------------

Для нахождения суммы простых порождающих делители чисел до 100000000:

.. code-block:: python

    result = sum_prime_generating_divisors(100000000)
    print(f"Сумма простых порождающих делителей чисел до 100000000: {result}")
"""
import math
from src.logger.logger import logger # Импорт logger для обработки ошибок


def is_prime(n: int) -> bool:
    """
    Проверяет, является ли число простым.

    :param n: Проверяемое число.
    :return: True, если число простое, иначе False.
    """
    if n <= 1:
        return False
    if n <= 3:
        return True
    if n % 2 == 0 or n % 3 == 0:
        return False
    i = 5
    while i * i <= n:
        if n % i == 0 or n % (i + 2) == 0:
            return False
        i += 6
    return True


def is_prime_generating_divisor(n: int) -> bool:
    """
    Проверяет, является ли число простым порождающим делители числом.

    :param n: Проверяемое число.
    :return: True, если число является простым порождающим делители числом, иначе False.
    """
    for d in range(1, int(math.sqrt(n)) + 1):
        if n % d == 0:
             # Проверка d
            if not is_prime(d + n // d):
                return False
            # Проверка n // d
            if d != n // d and not is_prime(n // d + d):
                return False
    return True


def sum_prime_generating_divisors(limit: int) -> int:
    """
    Вычисляет сумму всех простых порождающих делители чисел до заданного предела.

    :param limit: Верхний предел для поиска.
    :return: Сумма простых порождающих делители чисел.
    """
    total_sum = 0
    for n in range(1, limit + 1):
        if is_prime_generating_divisor(n):
             # Если число является простым порождающим делители числом, то оно добавляется к сумме
            total_sum += n
    return total_sum


if __name__ == '__main__':
    try:
       # Задаем предел для поиска простых порождающих делителей чисел
        limit = 100_000_000
        # Вычисляем сумму и выводим результат
        result = sum_prime_generating_divisors(limit)
        print(f"Сумма простых порождающих делителей чисел до {limit}: {result}")
    except Exception as e:
        logger.error(f"Произошла ошибка: {e}")
