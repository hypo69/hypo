"""
Модуль для решения задачи 645 проекта Эйлера.
=================================================

Этот модуль содержит функции для расчета наименьшего k,
при котором n + k является произведением двух и более различных простых чисел,
а также для определения, является ли число числом Идзуми.
"""
import math
from src.utils.jjson import j_loads, j_loads_ns
from src.logger.logger import logger
#  Импортируем необходимые модули: math для математических операций, src.utils.jjson для работы с JSON и src.logger.logger для логирования

def is_prime(n: int) -> bool:
    """
    Проверяет, является ли число простым.

    :param n: Целое число для проверки.
    :return: True, если число простое, False в противном случае.
    """
    if n <= 1:
        return False
    if n <= 3:
        return True
    if n % 2 == 0 or n % 3 == 0:
        return False
    for i in range(5, int(math.sqrt(n)) + 1, 6):
        if n % i == 0 or n % (i + 2) == 0:
            return False
    return True
    #  Определяем функцию is_prime, которая проверяет, является ли число простым

def prime_factorization(n: int) -> set[int]:
    """
    Находит множество простых множителей числа.

    :param n: Целое число для факторизации.
    :return: Множество простых множителей числа.
    """
    factors = set()
    d = 2
    while d * d <= n:
        if n % d == 0:
            factors.add(d)
            while n % d == 0:
                n //= d
        d += 1
    if n > 1:
        factors.add(n)
    return factors
    #  Определяем функцию prime_factorization, которая находит множество простых множителей числа

def f(n: int) -> int:
    """
    Находит наименьшее положительное целое число k, такое что n + k
    является произведением двух или более различных простых чисел.

    :param n: Целое число для которого ищется k.
    :return: Наименьшее k, удовлетворяющее условию.
    """
    k = 1
    while True:
        num = n + k
        factors = prime_factorization(num)
        if len(factors) >= 2:
            return k
        k += 1
        #  Определяем функцию f(n), которая находит наименьшее k, такое что n + k является произведением двух или более различных простых чисел

def is_izumi(n: int) -> bool:
    """
    Проверяет, является ли число n числом Идзуми.

    :param n: Целое число для проверки.
    :return: True, если число является числом Идзуми, False в противном случае.
    """
    try:
        result_f = f(n)
        return result_f >= math.sqrt(n)
    except Exception as ex:
        logger.error(f"Ошибка при проверке числа {n} на Идзуми: {ex}")
        return False
    #  Определяем функцию is_izumi, которая проверяет, является ли число Идзуми

def count_izumi_numbers(limit: int) -> int:
    """
    Подсчитывает количество чисел Идзуми, не превышающих заданный предел.

    :param limit: Верхний предел для поиска чисел Идзуми.
    :return: Количество чисел Идзуми, не превышающих предел.
    """
    count = 0
    try:
        for n in range(1, limit + 1):
            if is_izumi(n):
                count += 1
                logger.info(f"Найдено число Идзуми: {n}")
        logger.info(f"Общее количество чисел Идзуми до {limit}: {count}")
    except Exception as ex:
        logger.error(f"Ошибка при подсчете чисел Идзуми до {limit}: {ex}")
    return count
    #  Определяем функцию count_izumi_numbers, которая подсчитывает количество чисел Идзуми до заданного предела

if __name__ == '__main__':
    limit = 10**7
    #  Устанавливаем лимит для поиска чисел Идзуми
    result = count_izumi_numbers(limit)
    #  Вызываем функцию count_izumi_numbers для подсчета чисел Идзуми
    print(f"Количество чисел Идзуми до {limit}: {result}")
    #  Выводим результат на экран
