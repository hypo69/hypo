```python
"""
Модуль для хранения локаторов элементов магазина.
====================================================

Этот модуль содержит определения локаторов, используемых для поиска элементов на странице магазина.
"""
from src.utils.jjson import j_loads_ns  #  импортируем j_loads_ns для загрузки json с игнорированием ошибок
from src.logger.logger import logger

# Объявление переменной LOCATORS, содержащей локаторы магазина
LOCATORS: dict = {}

try:
    #  код исполняет открытие и загрузку JSON файла
    with open('hypotez/src/suppliers/visualdg/locators/store.json', 'r', encoding='utf-8') as f:
       # код исполняет загрузку данных из файла, используя j_loads_ns для обработки ошибок
        LOCATORS = j_loads_ns(f.read())
except FileNotFoundError as e:
    logger.error(f'Файл локаторов магазина не найден: {e}')
except Exception as e:
    logger.error(f'Ошибка при загрузке локаторов магазина: {e}')
    
#  Проверка загрузились ли локаторы
if not LOCATORS:
    logger.error("Локаторы магазина не загружены")

def get_store_locators() -> dict:
    """
    Возвращает локаторы элементов магазина.
    
    :return: Словарь, содержащий локаторы элементов магазина.
    :rtype: dict
    """
    return LOCATORS
```
