def normalize_fruits(fruits: str) -> str:
    """
    –ù–æ—Ä–º–∞–ª–∏–∑—É–µ—Ç —Å—Ç—Ä–æ–∫—É —Å —Ñ—Ä—É–∫—Ç–∞–º–∏, –ø—Ä–∏–≤–æ–¥—è –µ—ë –∫ –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–º—É –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–∏—é,

    :param fruits: –°—Ç—Ä–æ–∫–∞ —Å —Ñ—Ä—É–∫—Ç–∞–º–∏ (üçé, üçê, üçâ, üß∫).
    :type fruits: str
    :return: –°—Ç—Ä–æ–∫–∞ —Å –Ω–æ—Ä–º–∞–ª–∏–∑–æ–≤–∞–Ω–Ω—ã–º –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ–º —Ñ—Ä—É–∫—Ç–æ–≤.
    :rtype: str
    :raises TypeError: –ï—Å–ª–∏ –≤—Ö–æ–¥–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ –Ω–µ —è–≤–ª—è—é—Ç—Å—è —Å—Ç—Ä–æ–∫–æ–π.

    if not isinstance(fruits, str): # –ø—Ä–æ–≤–µ—Ä–∫–∞ —Ç–∏–ø–∞ –¥–∞–Ω–Ω—ã—Ö

        raise TypeError("–û–∂–∏–¥–∞–µ—Ç—Å—è —Å—Ç—Ä–æ–∫–∞")
    apples = fruits.count(APPLE) # –ø–æ–¥—Å—á–µ—Ç —è–±–ª–æ–∫

    melons %= 2

    # –°–æ–±–∏—Ä–∞–µ–º —Å—Ç—Ä–æ–∫—É –æ–±—Ä–∞—Ç–Ω–æ, —Å–Ω–∞—á–∞–ª–∞ –∫–æ—Ä–∑–∏–Ω—ã, –ø–æ—Ç–æ–º –¥—ã–Ω–∏, –≥—Ä—É—à–∏, —è–±–ª–æ–∫–∏

        BASKET * baskets
        + MELON * melons
        + PEAR * pears
        + APPLE * apples
    )

    """

    melons2 = fruits2.count(MELON) # –ø–æ–¥—Å—á–µ—Ç –¥—ã–Ω—å
    baskets2 = fruits2.count(BASKET) # –ø–æ–¥—Å—á–µ—Ç –∫–æ—Ä–∑–∏–Ω


    # –í—Ä–µ–º–µ–Ω–Ω–æ–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–∏–µ –≤ –≤–∏–¥–µ –æ–±—â–µ–≥–æ –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ —è–±–ª–æ–∫
    total_apples1 = apples1 + pears1 * 3 + melons1 * 15 // 3 + baskets1 * 30
    total_apples2 = apples2 + pears2 * 3 + melons2 * 15 // 3 + baskets2 * 30

    if total_apples1 < total_apples2:
        return "–ù–µ–≤–æ–∑–º–æ–∂–Ω–æ –≤—ã—á–µ—Å—Ç—å"
    else:
        total_apples = total_apples1 - total_apples2

    # –í–æ–∑–≤—Ä–∞—â–∞–µ–º –Ω–æ—Ä–º–∞–ª–∏–∑–æ–≤–∞–Ω–Ω–æ–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–∏–µ —Å—É–º–º—ã —è–±–ª–æ–∫
    result_fruits = ""
    baskets = total_apples // 30
    result_fruits += BASKET * baskets
    total_apples %= 30
    melons = (total_apples * 3) // 15
    result_fruits += MELON * melons
    total_apples %= 15
    pears = total_apples // 3
    result_fruits += PEAR * pears
    total_apples %= 3
    result_fruits += APPLE * total_apples

    return normalize_fruits(result_fruits)



def bin_to_dec(binary: str) -> int:

        power += 1 # –£–≤–µ–ª–∏—á–∏–≤–∞–µ–º —Å—Ç–µ–ø–µ–Ω—å –¥–ª—è —Å–ª–µ–¥—É—é—â–µ–≥–æ —Ä–∞–∑—Ä—è–¥–∞
    return decimal # –í–æ–∑–≤—Ä–∞—â–∞–µ–º –¥–µ—Å—è—Ç–∏—á–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ


def dec_to_septenary(decimal: int) -> str:
    """
    –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ—Ç –¥–µ—Å—è—Ç–∏—á–Ω–æ–µ —á–∏—Å–ª–æ (—Ü–µ–ª–æ–µ) –≤ —Å–µ–º–µ—Ä–∏—á–Ω–æ–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–∏–µ (—Å—Ç—Ä–æ–∫—É).

    :return: –°–µ–º–µ—Ä–∏—á–Ω–æ–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–∏–µ —á–∏—Å–ª–∞ (—Å—Ç—Ä–æ–∫–∞).
    :rtype: str
    :raises TypeError: –ï—Å–ª–∏ –≤—Ö–æ–¥–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ –Ω–µ —è–≤–ª—è—é—Ç—Å—è —Ü–µ–ª—ã–º —á–∏—Å–ª–æ–º.
    :raises ValueError: –ï—Å–ª–∏ –≤—Ö–æ–¥–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ —è–≤–ª—è—é—Ç—Å—è –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–º —á–∏—Å–ª–æ–º.
    """

        logger.error(f"–û–∂–∏–¥–∞–µ—Ç—Å—è —Ü–µ–ª–æ–µ —á–∏—Å–ª–æ, –ø–æ–ª—É—á–µ–Ω–æ: {type(decimal)}")
        raise TypeError("–û–∂–∏–¥–∞–µ—Ç—Å—è —Ü–µ–ª–æ–µ —á–∏—Å–ª–æ")
    if decimal < 0: # –ø—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å
        logger.error(f"–í—Ö–æ–¥–Ω–æ–µ —á–∏—Å–ª–æ –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–º: {decimal}")
        raise ValueError("–î–µ—Å—è—Ç–∏—á–Ω–æ–µ —á–∏—Å–ª–æ –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–º")
    if decimal == 0: # –ï—Å–ª–∏ –¥–µ—Å—è—Ç–∏—á–Ω–æ–µ —á–∏—Å–ª–æ —Ä–∞–≤–Ω–æ 0
        return "0"  # –í–æ–∑–≤—Ä–∞—â–∞–µ–º —Å—Ç—Ä–æ–∫—É "0"
    septenary = ""  # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ–º —Å—Ç—Ä–æ–∫—É –¥–ª—è —Å–µ–º–µ—Ä–∏—á–Ω–æ–≥–æ —á–∏—Å–ª–∞
    while decimal > 0: # –ü–æ–∫–∞ –¥–µ—Å—è—Ç–∏—á–Ω–æ–µ —á–∏—Å–ª–æ –±–æ–ª—å—à–µ 0
        septenary = str(decimal % 7) + septenary  # –î–æ–±–∞–≤–ª—è–µ–º –æ—Å—Ç–∞—Ç–æ–∫ –æ—Ç –¥–µ–ª–µ–Ω–∏—è –Ω–∞ 7 –≤ –Ω–∞—á–∞–ª–æ —Å–µ–º–µ—Ä–∏—á–Ω–æ–π —Å—Ç—Ä–æ–∫–∏
        decimal = decimal // 7  # –¶–µ–ª–æ—á–∏—Å–ª–µ–Ω–Ω–æ –¥–µ–ª–∏–º –¥–µ—Å—è—Ç–∏—á–Ω–æ–µ —á–∏—Å–ª–æ –Ω–∞ 7
    return septenary # –í–æ–∑–≤—Ä–∞—â–∞–µ–º —Å–µ–º–µ—Ä–∏—á–Ω—É—é —Å—Ç—Ä–æ–∫—É



def hex_to_dec(hexadecimal: str) -> int:
    """
    –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ—Ç —à–µ—Å—Ç–Ω–∞–¥—Ü–∞—Ç–µ—Ä–∏—á–Ω–æ–µ —á–∏—Å–ª–æ (—Å—Ç—Ä–æ–∫—É) –≤ –¥–µ—Å—è—Ç–∏—á–Ω–æ–µ.

    :param hexadecimal: –®–µ—Å—Ç–Ω–∞–¥—Ü–∞—Ç–µ—Ä–∏—á–Ω–æ–µ —á–∏—Å–ª–æ –≤ –≤–∏–¥–µ —Å—Ç—Ä–æ–∫–∏.
    :type hexadecimal: str
    :return: –î–µ—Å—è—Ç–∏—á–Ω–æ–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–∏–µ —á–∏—Å–ª–∞ (—Ü–µ–ª–æ–µ —á–∏—Å–ª–æ).
    :rtype: int
    :raises TypeError: –ï—Å–ª–∏ –≤—Ö–æ–¥–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ –Ω–µ —è–≤–ª—è—é—Ç—Å—è —Å—Ç—Ä–æ–∫–æ–π.
    :raises ValueError: –ï—Å–ª–∏ –≤—Ö–æ–¥–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ —Å–æ–¥–µ—Ä–∂–∞—Ç –Ω–µ —à–µ—Å—Ç–Ω–∞–¥—Ü–∞—Ç–µ—Ä–∏—á–Ω—ã–µ —Å–∏–º–≤–æ–ª—ã.
    """
    if not isinstance(hexadecimal, str): # –ø—Ä–æ–≤–µ—Ä–∫–∞ —Ç–∏–ø–∞ –¥–∞–Ω–Ω—ã—Ö
        logger.error(f"–û–∂–∏–¥–∞–µ—Ç—Å—è —Å—Ç—Ä–æ–∫–∞, –ø–æ–ª—É—á–µ–Ω–æ: {type(hexadecimal)}")
        raise TypeError("–û–∂–∏–¥–∞–µ—Ç—Å—è —Å—Ç—Ä–æ–∫–∞")
    hex_digits = "0123456789ABCDEF"
    if not all(digit.upper() in hex_digits for digit in hexadecimal):  # –ø—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ hex —Å–∏–º–≤–æ–ª—ã
        logger.error(f"–í—Ö–æ–¥–Ω–∞—è —Å—Ç—Ä–æ–∫–∞ —Å–æ–¥–µ—Ä–∂–∏—Ç –Ω–µ–¥–æ–ø—É—Å—Ç–∏–º—ã–µ —Å–∏–º–≤–æ–ª—ã: {hexadecimal}")
        raise ValueError("–®–µ—Å—Ç–Ω–∞–¥—Ü–∞—Ç–µ—Ä–∏—á–Ω–æ–µ —á–∏—Å–ª–æ –¥–æ–ª–∂–Ω–æ —Å–æ–¥–µ—Ä–∂–∞—Ç—å —Ç–æ–ª—å–∫–æ —Ü–∏—Ñ—Ä—ã –æ—Ç 0 –¥–æ 9 –∏ –±—É–∫–≤—ã –æ—Ç A –¥–æ F")
    decimal = 0  # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ–º –¥–µ—Å—è—Ç–∏—á–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ
    power = 0 # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ–º —Å—Ç–µ–ø–µ–Ω—å 16 (–ø–æ–∫–∞–∑–∞—Ç–µ–ª—å —Ä–∞–∑—Ä—è–¥–∞)
    hex_values = {str(i): i for i in range(10)} # –∑–Ω–∞—á–µ–Ω–∏—è –¥–ª—è —Ü–∏—Ñ—Ä
    hex_values.update({'A': 10, 'B': 11, 'C': 12, 'D': 13, 'E': 14, 'F': 15}) # –¥–æ–±–∞–≤–ª—è–µ–º –∑–Ω–∞—á–µ–Ω–∏—è –¥–ª—è –±—É–∫–≤
    for digit in reversed(hexadecimal):  # –ò—Ç–µ—Ä–∏—Ä—É–µ–º—Å—è –ø–æ —Ü–∏—Ñ—Ä–∞–º —à–µ—Å—Ç–Ω–∞–¥—Ü–∞—Ç–µ—Ä–∏—á–Ω–æ–≥–æ —á–∏—Å–ª–∞ –≤ –æ–±—Ä–∞—Ç–Ω–æ–º –ø–æ—Ä—è–¥–∫–µ
        decimal += hex_values[digit.upper()] * (16 ** power)  # –î–æ–±–∞–≤–ª—è–µ–º —Ü–∏—Ñ—Ä—É * 16 –≤ —Å—Ç–µ–ø–µ–Ω–∏ —Ä–∞–∑—Ä—è–¥–∞
        power += 1 # –£–≤–µ–ª–∏—á–∏–≤–∞–µ–º —Å—Ç–µ–ø–µ–Ω—å –¥–ª—è —Å–ª–µ–¥—É—é—â–µ–≥–æ —Ä–∞–∑—Ä—è–¥–∞
    return decimal # –í–æ–∑–≤—Ä–∞—â–∞–µ–º –¥–µ—Å—è—Ç–∏—á–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ

def calculate_day_of_week(start_day: int, days_passed: float) -> int:
    """
    –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ—Ç –¥–µ–Ω—å –Ω–µ–¥–µ–ª–∏ –ø–æ—Å–ª–µ –∑–∞–¥–∞–Ω–Ω–æ–≥–æ –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ –¥–Ω–µ–π.

    :param start_day: –ù–∞—á–∞–ª—å–Ω—ã–π –¥–µ–Ω—å –Ω–µ–¥–µ–ª–∏ (0 - –ø–æ–Ω–µ–¥–µ–ª—å–Ω–∏–∫, 6 - –≤–æ—Å–∫—Ä–µ—Å–µ–Ω—å–µ).
    :type start_day: int
    :param days_passed: –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –ø—Ä–æ—à–µ–¥—à–∏—Ö –¥–Ω–µ–π.
    :type days_passed: float
    :return: –î–µ–Ω—å –Ω–µ–¥–µ–ª–∏ –ø–æ—Å–ª–µ –∑–∞–¥–∞–Ω–Ω–æ–≥–æ –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ –¥–Ω–µ–π (0 - –ø–æ–Ω–µ–¥–µ–ª—å–Ω–∏–∫, 6 - –≤–æ—Å–∫—Ä–µ—Å–µ–Ω—å–µ).
    :rtype: int
    :raises TypeError: –ï—Å–ª–∏ start_day –Ω–µ —è–≤–ª—è–µ—Ç—Å—è —Ü–µ–ª—ã–º —á–∏—Å–ª–æ–º –∏–ª–∏ days_passed –Ω–µ —è–≤–ª—è–µ—Ç—Å—è —á–∏—Å–ª–æ–º.
    :raises ValueError: –ï—Å–ª–∏ start_day –Ω–µ –Ω–∞—Ö–æ–¥–∏—Ç—Å—è –≤ –¥–∏–∞–ø–∞–∑–æ–Ω–µ –æ—Ç 0 –¥–æ 6.
    """
    if not isinstance(start_day, int):  # –ø—Ä–æ–≤–µ—Ä–∫–∞ —Ç–∏–ø–∞ –¥–∞–Ω–Ω—ã—Ö
        logger.error(f"–û–∂–∏–¥–∞–µ—Ç—Å—è —Ü–µ–ª–æ–µ —á–∏—Å–ª–æ, –ø–æ–ª—É—á–µ–Ω–æ: {type(start_day)}")
        raise TypeError("–ù–∞—á–∞–ª—å–Ω—ã–π –¥–µ–Ω—å –Ω–µ–¥–µ–ª–∏ –¥–æ–ª–∂–µ–Ω –±—ã—Ç—å —Ü–µ–ª—ã–º —á–∏—Å–ª–æ–º")
    if not isinstance(days_passed, (int, float)): # –ø—Ä–æ–≤–µ—Ä–∫–∞ —Ç–∏–ø–∞ –¥–∞–Ω–Ω—ã—Ö
        logger.error(f"–û–∂–∏–¥–∞–µ—Ç—Å—è —á–∏—Å–ª–æ, –ø–æ–ª—É—á–µ–Ω–æ: {type(days_passed)}")
        raise TypeError("–ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –ø—Ä–æ—à–µ–¥—à–∏—Ö –¥–Ω–µ–π –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å —á–∏—Å–ª–æ–º")
    if not (0 <= start_day <= 6): # –ø—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –¥–æ–ø—É—Å—Ç–∏–º—ã–π –¥–∏–∞–ø–∞–∑–æ–Ω
        logger.error(f"–ù–∞—á–∞–ª—å–Ω—ã–π –¥–µ–Ω—å –Ω–µ–¥–µ–ª–∏ –¥–æ–ª–∂–µ–Ω –±—ã—Ç—å —Ü–µ–ª—ã–º —á–∏—Å–ª–æ–º –æ—Ç 0 –¥–æ 6 (–ø–Ω-–≤—Å), –ø–æ–ª—É—á–µ–Ω–æ: {start_day}")
        raise ValueError("–ù–∞—á–∞–ª—å–Ω—ã–π –¥–µ–Ω—å –Ω–µ–¥–µ–ª–∏ –¥–æ–ª–∂–µ–Ω –±—ã—Ç—å —Ü–µ–ª—ã–º —á–∏—Å–ª–æ–º –æ—Ç 0 –¥–æ 6 (–ø–Ω-–≤—Å)")

    days_passed = int(days_passed)
    new_day = (start_day + days_passed) % 7
    return new_day

        logger.error(f"–û–∂–∏–¥–∞–µ—Ç—Å—è —Ü–µ–ª–æ–µ —á–∏—Å–ª–æ, –ø–æ–ª—É—á–µ–Ω–æ: {type(day_number)}")
