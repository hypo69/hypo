```python
"""
Модуль для решения задачи 127 Project Euler: ABC-хиты.
======================================================

Этот модуль содержит функции для вычисления наибольшего общего делителя, радикала числа,
и проверки условий abc-хита для поиска и суммирования таких троек чисел.
"""
import math
from src.logger.logger import logger #импортируем logger


def gcd(a: int, b: int) -> int:
    """
    Вычисляет наибольший общий делитель (НОД) двух целых чисел.

    :param a: Первое целое число.
    :param b: Второе целое число.
    :return: Наибольший общий делитель a и b.
    """
    return math.gcd(a, b)

def rad(n: int) -> int:
    """
    Вычисляет радикал числа n, то есть произведение его простых делителей.

    :param n: Целое число, для которого вычисляется радикал.
    :return: Радикал числа n.
    """
    if n == 1:
        return 1
    res = 1
    i = 2
    temp = n
    while i * i <= temp:
        if temp % i == 0:
            res *= i
            while temp % i == 0:
                temp //= i
        i += 1
    if temp > 1:
        res *= temp
    return res


def is_abc_hit(a: int, b: int, c: int) -> bool:
    """
    Проверяет, является ли тройка чисел (a, b, c) abc-хитом.

    :param a: Первое целое число.
    :param b: Второе целое число.
    :param c: Третье целое число.
    :return: True, если тройка (a, b, c) является abc-хитом, иначе False.
    """
    if gcd(a, b) != 1 or gcd(a, c) != 1 or gcd(b, c) != 1:
        return False
    if a >= b:
        return False
    if a + b != c:
        return False
    if rad(a * b * c) >= c:
        return False
    return True


def sum_abc_hits(limit: int) -> int:
    """
    Вычисляет сумму abc-хитов для c <= limit.

    :param limit: Верхняя граница для c.
    :return: Сумма всех abc-хитов, найденных в пределах лимита.
    """
    total_sum = 0
    for c in range(2, limit + 1):
        for a in range(1, c // 2 + 1):
            b = c - a
            if is_abc_hit(a, b, c):
                total_sum += c
    return total_sum


if __name__ == '__main__':
    limit = 120000
    result = sum_abc_hits(limit)
    print(f"Сумма abc-хитов для c ≤ {limit}: {result}")
    #  Код выполняет расчет суммы abc хитов для лимита 120000 и выводит результат
```
