"""
Модуль для решения задачи 133 Project Euler.
=========================================================================================

Этот модуль содержит функции для определения репьюнитов, проверки чисел на простоту
и поиска суммы "плохих" простых чисел, которые не делят репьюниты с простым k.
"""
import math
from src.logger.logger import logger


LIMIT: int = 100000


def repunit(k: int) -> int:
    """
    Вычисляет репьюнит R(k) = (10^k - 1) / 9.

    :param k: Длина репьюнита.
    :type k: int
    :return: Значение репьюнита R(k).
    :rtype: int
    """
    return (10**k - 1) // 9


def is_prime(n: int) -> bool:
    """
    Проверяет, является ли число простым.

    :param n: Проверяемое число.
    :type n: int
    :return: `True`, если число простое, иначе `False`.
    :rtype: bool
    """
    if n <= 1:
        return False
    for i in range(2, int(math.sqrt(n)) + 1):
        if n % i == 0:
            return False
    return True


def find_smallest_k(p: int) -> int:
    """
    Находит наименьшее k, такое что R(k) делится на p.

    :param p: Простое число, делитель репьюнита.
    :type p: int
    :return: Наименьшее k, такое что R(k) делится на p, или -1, если такого k нет.
    :rtype: int
    """
    for k in range(1, p):
        if repunit(k) % p == 0:
            return k
    return -1


def is_bad_prime(p: int) -> bool:
    """
    Проверяет, является ли простое число "плохим" простым.

    Простое число p является "плохим", если наименьшее k, такое что R(k) делится на p,
    не является простым числом.

    :param p: Проверяемое простое число.
    :type p: int
    :return: `True`, если простое число "плохое", иначе `False`.
    :rtype: bool
    """
    smallest_k = find_smallest_k(p)
    if smallest_k == -1:
        return False
    return not is_prime(smallest_k)


def sum_of_bad_primes(limit: int) -> int:
    """
    Вычисляет сумму всех "плохих" простых чисел меньше заданного предела.

    :param limit: Верхняя граница диапазона для поиска простых чисел.
    :type limit: int
    :return: Сумма всех "плохих" простых чисел меньше `limit`.
    :rtype: int
    """
    bad_primes_sum = 0
    for number in range(2, limit):
        if is_prime(number):
            if is_bad_prime(number):
                bad_primes_sum += number
    return bad_primes_sum


if __name__ == '__main__':
    # Пример использования
    try:
        result = sum_of_bad_primes(LIMIT)
        print(result)
    except Exception as e:
        logger.error(f'Произошла ошибка при выполнении: {e}')
