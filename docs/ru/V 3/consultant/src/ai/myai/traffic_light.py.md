```python
## \file /src/ai/myai/traffic_light.py
# -*- coding: utf-8 -*-

"""
Модуль для управления логикой светофора.
========================================

Этот модуль предназначен для моделирования и управления поведением светофора.
Он предоставляет возможность определять текущее состояние светофора,
переключать его между различными состояниями (красный, желтый, зеленый)
и логировать изменения состояния.

Пример использования:
--------------------

>>> traffic_light = TrafficLight()
>>> traffic_light.get_state()
'red'
>>> traffic_light.switch_to('green')
>>> traffic_light.get_state()
'green'
"""

from src.logger import logger  # Импорт logger из src.logger


class TrafficLight:
    """
    Класс, представляющий светофор.
    """

    def __init__(self):
        """
        Инициализация светофора с начальным состоянием "красный".
        """
        self.state = 'red'  # Начальное состояние светофора

    def get_state(self) -> str:
        """
        Возвращает текущее состояние светофора.

        Returns:
            str: Текущее состояние светофора.
        """
        return self.state

    def switch_to(self, new_state: str) -> None:
        """
        Переключает светофор в новое состояние.

        Args:
            new_state (str): Новое состояние светофора ('red', 'yellow', 'green').

        Raises:
            ValueError: Если `new_state` не является допустимым состоянием.
        """
        if new_state not in ('red', 'yellow', 'green'):
            logger.error(f'Invalid state: {new_state}')  # Логирование некорректного состояния
            raise ValueError(f'Invalid state: {new_state}')

        logger.info(f'Switching from {self.state} to {new_state}')  # Логирование переключения состояния
        self.state = new_state


if __name__ == '__main__':
    traffic_light = TrafficLight()
    print(f'Initial state: {traffic_light.get_state()}')
    traffic_light.switch_to('green')
    print(f'New state: {traffic_light.get_state()}')
```
