```python
## \file /src/utils/powershell/examples/pprint/header.py
# -*- coding: utf-8 -*-

#! .pyenv/bin/python3

"""
Модуль для определения корневого пути проекта и добавления его в sys.path.
========================================================================

Этот модуль определяет корневой путь проекта `hypotez` и добавляет его в `sys.path`,
чтобы обеспечить возможность импорта других модулей проекта.

Пример использования:
----------------------

>>> from src.utils.powershell.examples.pprint.header import __root__
>>> print(__root__)
/path/to/hypotez
"""

import sys
import os
from pathlib import Path
from src.logger import logger  # Import logger

try:
    # Определяем корневой путь проекта 'hypotez'
    current_path: str = os.getcwd()
    root_index: int = current_path.rfind('hypotez') + 7
    __root__: Path = Path(current_path[:root_index])

    # Добавляем корневой путь в sys.path для возможности импорта модулей
    sys.path.append(str(__root__))
    logger.info(f'Root path added to sys.path: {__root__}')

except Exception as e:
    logger.error(f'Error while setting up root path: {e}', exc_info=True)
    __root__: Path = Path('')  # Ensure __root__ is always defined, even on error
```
