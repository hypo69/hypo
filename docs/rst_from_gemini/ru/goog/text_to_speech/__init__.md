"""
Модуль для работы с Google Text-to-Speech.
"""
MODE = 'debug'  # Режим работы (отладка/производство)


import pyttsx3
from gtts import gTTS


class TTS:
    """
    Класс для работы с Google Text-to-Speech.
    """
    def __init__(self, voice='en'):
        """
        Инициализирует объект TTS.

        Args:
            voice (str): Код языка для голосового синтеза.
                       По умолчанию 'en' (английский).
        """

        try:
            self.tts = pyttsx3.init()
            self.voices = self.tts.getProperty('voices')
