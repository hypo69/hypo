"""
Модуль содержит список прокси-серверов для использования вебдрайвером.
Прокси-сервера представлены в формате `socks5://ip:порт` или `http://ip:порт` для разных протоколов.
"""
from src.utils.jjson import j_loads, j_loads_ns  # Импортируем необходимые функции для парсинга JSON
from src.logger.logger import logger  # Импортируем logger для логирования

# proxies_list = [] # Переменная для хранения списка прокси (удалено, так как она не использовалась)

def load_proxies(proxies_file: str) -> list:
    """
    Загружает список прокси-серверов из файла.

    :param proxies_file: Путь к файлу с прокси.
    :type proxies_file: str
    :raises FileNotFoundError: Если файл не найден.
    :raises ValueError: Если содержимое файла не соответствует ожидаемому формату.
    :return: Список прокси-серверов.
    :rtype: list
    """
    try:
        # код исполняет чтение файла с прокси-серверами
        with open(proxies_file, "r") as file:
            proxies = j_loads(file.read())
            # проверка, что полученный список является списком строк
            if not all(isinstance(proxy, str) for proxy in proxies):
                raise ValueError("Прокси-сервера не в правильном формате.")

            return proxies

    except FileNotFoundError as e:
        logger.error("Файл прокси-серверов не найден.", e)
        return []  # Возвращаем пустой список, если файл не найден
    except Exception as ex:
        logger.error("Ошибка при загрузке прокси-серверов", ex)
        return []  # Возвращаем пустой список при ошибках
