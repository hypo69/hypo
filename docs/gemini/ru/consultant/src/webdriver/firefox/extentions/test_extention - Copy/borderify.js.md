"""
Модуль для добавления красной рамки вокруг элемента <body>.
=========================================================================================

Этот модуль добавляет красную рамку вокруг элемента <body> на веб-странице.

Пример использования
--------------------

Модуль автоматически применяется к текущей веб-странице при загрузке расширения.
"""
from src.logger.logger import logger

try:
    # Проверка существования элемента body
    if document.body:
        # Установка границы для элемента body
        document.body.style.border = "5px solid red";
        # Логирование успешной установки границы
        logger.info("Граница установлена успешно");
    else:
         # Логирование ошибки, если body отсутствует
        logger.error("Элемент body не найден");
except Exception as ex:
    # Логирование ошибок при возникновении исключения
    logger.error(f"Произошла ошибка при установке границы: {ex}");
