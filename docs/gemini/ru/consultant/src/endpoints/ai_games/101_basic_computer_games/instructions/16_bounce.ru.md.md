```python
"""
Модуль, имитирующий игру "BOUNCE" (Отскок мяча).
=================================================

Этот модуль моделирует движение мяча, отскакивающего от поверхности, и отображает его траекторию.
"""
import time # импортируем модуль time для управления временем
from src.logger.logger import logger # импортируем logger для логирования ошибок

def get_float_input(prompt: str) -> float:
    """
    Запрашивает у пользователя ввод вещественного числа.

    :param prompt: Сообщение для пользователя.
    :return: Введенное пользователем вещественное число.
    :raises ValueError: Если ввод пользователя не является числом.
    """
    while True:
        try:
            value = float(input(prompt)) # код исполняет запрос ввода значения с консоли и преобразует в тип float
            return value # возвращаем значение
        except ValueError as ex:
            logger.error(f"Ошибка ввода: {ex}, введите корректное число.", exc_info=True) # логируем ошибку если ввод не корректный

def get_positive_float_input(prompt: str) -> float:
    """
    Запрашивает у пользователя ввод положительного вещественного числа.

    :param prompt: Сообщение для пользователя.
    :return: Введенное пользователем положительное вещественное число.
    :raises ValueError: Если ввод пользователя не является положительным числом.
    """
    while True:
        value = get_float_input(prompt) # код исполняет запрос ввода float числа через функцию get_float_input
        if value > 0: # проверка на положительное значение
            return value # возвращаем положительное значение
        else:
            logger.error(f"Ошибка ввода: Введите положительное число.")# логируем ошибку если число не положительное

def bounce_game():
    """
    Запускает игру "BOUNCE".

    Эта функция моделирует движение мяча, отскакивающего от поверхности, на основе
    введенных пользователем параметров.
    """
    print("Добро пожаловать в игру BOUNCE!") # выводим приветствие
    initial_speed = get_positive_float_input("Введите начальную скорость мяча (футы в секунду): ") # код исполняет запрос ввода начальной скорости через функцию get_positive_float_input
    elasticity_coefficient = get_float_input("Введите коэффициент упругости (например, 0.85 для супербола): ") # код исполняет запрос ввода коэффициента упругости через функцию get_float_input
    time_interval = get_positive_float_input("Введите интервал времени (например, 0.1 секунды): ") # код исполняет запрос ввода интервала времени через функцию get_positive_float_input

    time_elapsed = 0 # устанавливаем время = 0
    position = 0 # устанавливаем позицию = 0
    speed = initial_speed # устанавливаем скорость = начальной скорости

    print("Траектория мяча:") # выводим заголовок траектории
    while True: # запускаем бесконечный цикл
        position += speed * time_interval # изменяем позицию
        time_elapsed += time_interval # изменяем время
        print(f"Время: {time_elapsed:.1f} сек, Позиция: {position:.1f} футов") # выводим значения

        if position <= 0: # если позиция меньше или равна нулю
            speed = -speed * elasticity_coefficient # изменяем направление скорости и уменьшаем её в зависимости от коэф упругости

        choice = input("Хотите изменить параметры или выйти? (да/нет): ").lower() # запрашиваем ввод пользователя
        if choice == "да": # если ввод да, то запускаем новую игру с новыми параметрами
            bounce_game() # рекурсивный вызов
            return # выход из текущего вызова
        elif choice == "нет": # если ввод нет, то завершаем игру
            print("До свидания!") # выводим сообщение о завершении игры
            return # выход из функции

        time.sleep(time_interval) # приостанавливаем выполнение

if __name__ == "__main__":
    bounce_game() # запускаем игру
```
