В этой игре вы — капитан подводной лодки, а ваш противник — капитан боевого корабля. Цель игры — потопить боевой корабль, стреляя торпедами. Однако, если вы промахнётесь хотя бы один раз, боевой корабль откроет огонь в ответ, и игра превратится в обмен ударами до тех пор, пока один из игроков не потопит другого.

   - Ваша подводная лодка находится на определённом расстоянии от боевого корабля (например, 5136.84 ярдов), и вам нужно рассчитать правильный угол для попадания.

### Реализация

#### Расчет траектории торпеды:
- Траектория рассчитывается на основе угла и начального расстояния до цели.
- Для простоты, можно использовать линейную траекторию (без учета гравитации).
- Проверка попадания будет осуществляться на основе заданного радиуса попадания вокруг цели.

# Пример реализации (основная логика)
import math
import random

class Game:
    """
    Класс, представляющий игру "BOAT".

    :ivar ship_speed: скорость боевого корабля.
    :vartype ship_speed: float
    :ivar distance: расстояние до боевого корабля.
    :vartype distance: float
    :ivar ship_health: здоровье боевого корабля.
    :vartype ship_health: int
    :ivar is_vulnerable: флаг, указывающий на уязвимое место.
    :vartype is_vulnerable: bool
    """
    def __init__(self):
        """
        Инициализирует игру.
        """
        self.ship_speed = 17.18 # скорость корабля
        self.distance = 5136.84 # расстояние до корабля
        self.ship_health = 3 # здоровье корабля (2 попадания в уязвимое место или 1 уязвимое + 2 обычных)
        self.is_vulnerable = False

    def calculate_hit(self, torpedo_angle: float) -> bool:
        """
        Выполняет расчет попадания торпеды в цель.

        :param torpedo_angle: угол выстрела торпеды.
        :type torpedo_angle: float
        :return: `True`, если торпеда попала в уязвимое место, `False` в противном случае.
        :rtype: bool
        """
        # Логика расчета попадания в уязвимое место
        target_angle = 45 # Для примера, уязвимый угол 45
        hit = abs(torpedo_angle - target_angle) < 10 # Попадание если угол отличается от 45 менее чем на 10 градусов
        if hit:
            self.is_vulnerable = random.random() < 0.5 # Шанс 50% на попадание в уязвимое место
        return hit

    def play_turn(self):
       """
       Выполняет один ход игры.

        Запрашивает у игрока угол, проводит расчет попадания и обрабатывает последствия.
       """
       try:
        torpedo_angle = float(input("Введите угол для выстрела: "))
       except ValueError:
           print("Некорректный ввод. Пожалуйста, введите число.")
           return
       hit = self.calculate_hit(torpedo_angle)
       if hit and self.is_vulnerable:
           print("Торпеда выпущена... Попадание в уязвимое место!")
           self.ship_health -= 2
       elif hit:
           print("Торпеда выпущена... Попадание!")
           self.ship_health -= 1
       else:
           print("Вы промахнулись. Боевой корабль открывает огонь!")
           # Код ответного огня противника
       if self.ship_health <= 0:
          return True
       return False


    def start_game(self):
        """
        Запускает игру и управляет игровым процессом.
        """
        print("Добро пожаловать в игру BOAT!")
        print(f"Скорость боевого корабля установлена на {self.ship_speed} узлов.")
        print(f"Расстояние от вас до боевого корабля: {self.distance} ярдов.")
        print("Вам необходимо рассчитать угол для выстрела.")
        while True:
            game_over = self.play_turn()
            if game_over:
                print("Вы потопили боевой корабль!")
                print("Поздравляем, вы выиграли!")
                break
            if self.ship_health <= 0:
                break



if __name__ == "__main__":
    game = Game()
    game.start_game()
    while True:
      play_again = input("Хотите сыграть снова? (да/нет): ")
      if play_again.lower() != "да":
          break
      game = Game()
      game.start_game()
