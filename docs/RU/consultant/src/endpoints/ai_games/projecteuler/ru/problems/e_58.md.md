"""
Модуль для решения задачи №58 Project Euler.
================================================

Этот модуль содержит функцию для вычисления длины стороны спирали,
при которой отношение простых чисел на диагоналях становится меньше 10%.

Функция :func:`find_spiral_side` реализует алгоритм для нахождения
искомой длины.

Пример использования
--------------------

.. code-block:: python

    side_length = find_spiral_side()
    print(f"Длина стороны спирали: {side_length}")
"""
from src.logger.logger import logger #  импортируем logger для логирования

def is_prime(number: int) -> bool:
    """
    Проверяет, является ли число простым.

    :param number: Число для проверки.
    :return: True, если число простое, иначе False.
    """
    if number < 2:
        return False
    if number == 2:
        return True
    if number % 2 == 0:
        return False
    for i in range(3, int(number**0.5) + 1, 2):
        if number % i == 0:
            return False
    return True

def find_spiral_side() -> int:
    """
    Находит длину стороны спирали, при которой отношение простых чисел на диагоналях меньше 10%.

    :return: Длина стороны спирали.
    """
    primes_count = 0
    total_numbers = 1
    number = 1
    side_length = 1
    while True:
        side_length += 2
        for _ in range(4):
            number += side_length - 1
            if is_prime(number):
                primes_count += 1
            total_numbers += 1
        if primes_count / total_numbers < 0.1:
            return side_length
