8.  –û–±–µ—Å–ø–µ—á–∏—Ç—å —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏–µ —Å—Ç–∏–ª—é –∫–æ–¥–∞ –∏ —Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏—é, –ø—Ä–∏–Ω—è—Ç–æ–º—É –≤ –ø—Ä–æ–µ–∫—Ç–µ.

**–£–ª—É—á—à–µ–Ω–Ω—ã–π –∫–æ–¥**

```python
"""
–ú–æ–¥—É–ª—å, –¥–µ–º–æ–Ω—Å—Ç—Ä–∏—Ä—É—é—â–∏–π –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ –º–Ω–æ–∂–µ—Å—Ç–≤ (–Ω–∞–±–æ—Ä–æ–≤) –≤ Python.
=========================================================================================

–ú–æ–¥—É–ª—å –ø—Ä–µ–¥–æ—Å—Ç–∞–≤–ª—è–µ—Ç –ø—Ä–∏–º–µ—Ä—ã —Å–æ–∑–¥–∞–Ω–∏—è –º–Ω–æ–∂–µ—Å—Ç–≤, –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è –æ—Å–Ω–æ–≤–Ω—ã—Ö –æ–ø–µ—Ä–∞—Ü–∏–π –Ω–∞–¥ –Ω–∏–º–∏
–∏ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è –¥–ª—è —Ä–µ—à–µ–Ω–∏—è –ø—Ä–æ—Å—Ç—ã—Ö –∑–∞–¥–∞—á.

–ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è
--------------------

–ü—Ä–∏–º–µ—Ä —Å–æ–∑–¥–∞–Ω–∏—è –∏ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è –º–Ω–æ–∂–µ—Å—Ç–≤ —Ñ—Ä—É–∫—Ç–æ–≤:

.. code-block:: python

   from typing import Set

   fruits_set_A = create_fruit_set("üçéüçê")
   fruits_set_B = create_fruit_set("üçêüçâ")
   union_result = fruits_set_A | fruits_set_B
   print(display_set(union_result))
"""
from typing import Set # –ò–º–ø–æ—Ä—Ç Set –∏–∑ –º–æ–¥—É–ª—è typing –¥–ª—è –∞–Ω–Ω–æ—Ç–∞—Ü–∏–∏ —Ç–∏–ø–æ–≤
from src.logger.logger import logger # –ò–º–ø–æ—Ä—Ç logger –¥–ª—è –ª–æ–≥–∏—Ä–æ–≤–∞–Ω–∏—è –æ—à–∏–±–æ–∫

def create_fruit_set(fruit_string: str) -> Set[str]:
    """
    –°–æ–∑–¥–∞–µ—Ç –º–Ω–æ–∂–µ—Å—Ç–≤–æ —Ñ—Ä—É–∫—Ç–æ–≤ –∏–∑ —Å—Ç—Ä–æ–∫–∏.

    :param fruit_string: –°—Ç—Ä–æ–∫–∞ —Å —Ñ—Ä—É–∫—Ç–∞–º–∏ (üçé, üçê, üçâ, üß∫).
    :type fruit_string: str
    :raises ValueError: –ï—Å–ª–∏ —Å—Ç—Ä–æ–∫–∞ —Å–æ–¥–µ—Ä–∂–∏—Ç –Ω–µ–¥–æ–ø—É—Å—Ç–∏–º—ã–µ —Å–∏–º–≤–æ–ª—ã.
    :return: –ú–Ω–æ–∂–µ—Å—Ç–≤–æ —É–Ω–∏–∫–∞–ª—å–Ω—ã—Ö —Ñ—Ä—É–∫—Ç–æ–≤.
    :rtype: Set[str]
    """
    if not isinstance(fruit_string, str): # –ü—Ä–æ–≤–µ—Ä–∫–∞ —Ç–∏–ø–∞ –¥–ª—è fruit_string
        logger.error(f"–ù–µ–≤–µ—Ä–Ω—ã–π —Ç–∏–ø –¥–∞–Ω–Ω—ã—Ö –¥–ª—è fruit_string: {type(fruit_string)}")  # –õ–æ–≥–∏—Ä–æ–≤–∞–Ω–∏–µ –æ—à–∏–±–∫–∏ —Ç–∏–ø–∞ –¥–∞–Ω–Ω—ã—Ö
        raise TypeError("fruit_string –¥–æ–ª–∂–µ–Ω –±—ã—Ç—å —Å—Ç—Ä–æ–∫–æ–π") # –í—ã–±—Ä–∞—Å—ã–≤–∞–µ–º –æ—à–∏–±–∫—É TypeError –µ—Å–ª–∏ fruit_string –Ω–µ —Å—Ç—Ä–æ–∫–∞
    if not all(fruit in ["üçé", "üçê", "üçâ", "üß∫"] for fruit in fruit_string):
        logger.error(f"–ù–µ–¥–æ–ø—É—Å—Ç–∏–º—ã–µ —Å–∏–º–≤–æ–ª—ã –≤ —Å—Ç—Ä–æ–∫–µ: {fruit_string}") # –õ–æ–≥–∏—Ä–æ–≤–∞–Ω–∏–µ –æ—à–∏–±–∫–∏ –Ω–µ–¥–æ–ø—É—Å—Ç–∏–º—ã—Ö —Å–∏–º–≤–æ–ª–æ–≤
        raise ValueError("–°—Ç—Ä–æ–∫–∞ –º–æ–∂–µ—Ç —Å–æ–¥–µ—Ä–∂–∞—Ç—å —Ç–æ–ª—å–∫–æ —Å–∏–º–≤–æ–ª—ã üçé, üçê, üçâ, üß∫") # –í—ã–±—Ä–∞—Å—ã–≤–∞–µ–º –æ—à–∏–±–∫—É ValueError –µ—Å–ª–∏ –≤ fruit_string –µ—Å—Ç—å –Ω–µ–¥–æ–ø—É—Å—Ç–∏–º—ã–µ —Å–∏–º–≤–æ–ª—ã
    return set(fruit_string)  # –ò—Å–ø–æ–ª—å–∑—É–µ–º set() –¥–ª—è —Å–æ–∑–¥–∞–Ω–∏—è –º–Ω–æ–∂–µ—Å—Ç–≤–∞

def display_set(fruit_set: Set[str]) -> str:
    """
    –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ—Ç –º–Ω–æ–∂–µ—Å—Ç–≤–æ —Ñ—Ä—É–∫—Ç–æ–≤ –≤ —Å—Ç—Ä–æ–∫—É –¥–ª—è –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏—è.

    :param fruit_set: –ú–Ω–æ–∂–µ—Å—Ç–≤–æ —Ñ—Ä—É–∫—Ç–æ–≤.
    :type fruit_set: Set[str]
    :return: –°—Ç—Ä–æ–∫–∞ –¥–ª—è –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏—è.
    :rtype: str
    """
    if not isinstance(fruit_set, set): # –ü—Ä–æ–≤–µ—Ä–∫–∞ —Ç–∏–ø–∞ –¥–ª—è fruit_set
        logger.error(f"–ù–µ–≤–µ—Ä–Ω—ã–π —Ç–∏–ø –¥–∞–Ω–Ω—ã—Ö –¥–ª—è fruit_set: {type(fruit_set)}") # –õ–æ–≥–∏—Ä–æ–≤–∞–Ω–∏–µ –æ—à–∏–±–∫–∏ —Ç–∏–ø–∞ –¥–∞–Ω–Ω—ã—Ö
        return "" # –í–æ–∑–≤—Ä–∞—â–∞–µ–º –ø—É—Å—Ç—É—é —Å—Ç—Ä–æ–∫—É –µ—Å–ª–∏ fruit_set –Ω–µ –º–Ω–æ–∂–µ—Å—Ç–≤–æ
    return "{" + ", ".join(fruit_set) + "}" # –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º –º–Ω–æ–∂–µ—Å—Ç–≤–æ –≤ —Å—Ç—Ä–æ–∫—É –¥–ª—è –≤—ã–≤–æ–¥–∞


# –°–æ–∑–¥–∞–µ–º –º–Ω–æ–∂–µ—Å—Ç–≤–∞ —Ñ—Ä—É–∫—Ç–æ–≤
fruits_set_A = create_fruit_set("üçéüçê")  # –ú–Ω–æ–∂–µ—Å—Ç–≤–æ A: {üçé, üçê}
fruits_set_B = create_fruit_set("üçêüçâ")  # –ú–Ω–æ–∂–µ—Å—Ç–≤–æ B: {üçê, üçâ}
fruits_set_C = create_fruit_set("üçéüçêüçâ")  # –ú–Ω–æ–∂–µ—Å—Ç–≤–æ C: {üçé, üçê, üçâ}
fruits_set_D = create_fruit_set("üß∫")  # –ú–Ω–æ–∂–µ—Å—Ç–≤–æ D: {üß∫}

# –í—ã–≤–æ–¥–∏–º –º–Ω–æ–∂–µ—Å—Ç–≤–∞
print(f"–ú–Ω–æ–∂–µ—Å—Ç–≤–æ A: {display_set(fruits_set_A)}")
print(f"–ú–Ω–æ–∂–µ—Å—Ç–≤–æ B: {display_set(fruits_set_B)}")
print(f"–ú–Ω–æ–∂–µ—Å—Ç–≤–æ C: {display_set(fruits_set_C)}")
print(f"–ú–Ω–æ–∂–µ—Å—Ç–≤–æ D: {display_set(fruits_set_D)}")

# –û–±—ä–µ–¥–∏–Ω–µ–Ω–∏–µ –º–Ω–æ–∂–µ—Å—Ç–≤ (Union)
union_result = fruits_set_A | fruits_set_B
print(f"A ‚à™ B: {display_set(union_result)}")  # –†–µ–∑—É–ª—å—Ç–∞—Ç: {üçé, üçê, üçâ}

# –ü–µ—Ä–µ—Å–µ—á–µ–Ω–∏–µ –º–Ω–æ–∂–µ—Å—Ç–≤ (Intersection)
intersection_result = fruits_set_A & fruits_set_B
print(f"A ‚à© B: {display_set(intersection_result)}")  # –†–µ–∑—É–ª—å—Ç–∞—Ç: {üçê}

# –†–∞–∑–Ω–æ—Å—Ç—å –º–Ω–æ–∂–µ—Å—Ç–≤ (Difference)
difference_result_AB = fruits_set_A - fruits_set_B
print(f"A - B: {display_set(difference_result_AB)}")  # –†–µ–∑—É–ª—å—Ç–∞—Ç: {üçé}
difference_result_BA = fruits_set_B - fruits_set_A
print(f"B - A: {display_set(difference_result_BA)}")  # –†–µ–∑—É–ª—å—Ç–∞—Ç: {üçâ}

# –°–∏–º–º–µ—Ç—Ä–∏—á–Ω–∞—è —Ä–∞–∑–Ω–æ—Å—Ç—å –º–Ω–æ–∂–µ—Å—Ç–≤ (Symmetric Difference)
symmetric_difference_result = fruits_set_A ^ fruits_set_B
print(f"A ^ B: {display_set(symmetric_difference_result)}")  # –†–µ–∑—É–ª—å—Ç–∞—Ç: {üçé, üçâ}

# –ü–æ–¥–º–Ω–æ–∂–µ—Å—Ç–≤–æ (Subset)
subset_result1 = fruits_set_A <= fruits_set_C
print(f"A <= C: {subset_result1}")  # –†–µ–∑—É–ª—å—Ç–∞—Ç: True (A –ø–æ–¥–º–Ω–æ–∂–µ—Å—Ç–≤–æ C)
subset_result2 = fruits_set_A <= fruits_set_B
print(f"A <= B: {subset_result2}") # –†–µ–∑—É–ª—å—Ç–∞—Ç: False (A –Ω–µ –ø–æ–¥–º–Ω–æ–∂–µ—Å—Ç–≤–æ B)

# –ù–∞–¥–º–Ω–æ–∂–µ—Å—Ç–≤–æ (Superset)
superset_result1 = fruits_set_C >= fruits_set_A
print(f"C >= A: {superset_result1}")  # –†–µ–∑—É–ª—å—Ç–∞—Ç: True (C –Ω–∞–¥–º–Ω–æ–∂–µ—Å—Ç–≤–æ A)
superset_result2 = fruits_set_B >= fruits_set_A
print(f"B >= A: {superset_result2}")  # –†–µ–∑—É–ª—å—Ç–∞—Ç: False (B –Ω–µ –Ω–∞–¥–º–Ω–æ–∂–µ—Å—Ç–≤–æ A)

# –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –Ω–∞–ª–∏—á–∏–µ —ç–ª–µ–º–µ–Ω—Ç–∞
print(f"üçé in A: {'üçé' in fruits_set_A}")  # –†–µ–∑—É–ª—å—Ç–∞—Ç: True
print(f"üçâ in A: {'üçâ' in fruits_set_A}")  # –†–µ–∑—É–ª—å—Ç–∞—Ç: False

# –§—É–Ω–∫—Ü–∏–∏ `create_fruit_set` –∏ `display_set` —É–ø—Ä–æ—â–∞—é—Ç —Å–æ–∑–¥–∞–Ω–∏–µ –∏ –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏–µ –º–Ω–æ–∂–µ—Å—Ç–≤ —Ñ—Ä—É–∫—Ç–æ–≤.
```
