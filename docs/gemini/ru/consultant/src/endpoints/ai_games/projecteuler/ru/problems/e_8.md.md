from src.logger.logger import logger

def largest_product(number_string: str, series_length: int = 13) -> int:
    """
    Находит наибольшее произведение `series_length` последовательных цифр в строке `number_string`.

    :param number_string: Строка, содержащая цифры.
    :param series_length: Длина последовательности цифр для вычисления произведения.
    :return: Наибольшее произведение последовательности цифр.
    """
    try:
        # Код удаляет пробелы и переносы строк из исходной строки
        number_string = number_string.replace(' ', '').replace('\n', '')
        # Инициализируем `max_product` значением 1
        max_product = 1
         # Проверка на то что длина последовательности не больше длины строки
        if series_length > len(number_string):
            logger.error(f'Длина последовательности {series_length} больше длины строки {len(number_string)}')
            return 0
        # Код перебирает все последовательности заданной длины
        for i in range(len(number_string) - series_length + 1):
            product = 1
            # Код перемножает цифры в текущей последовательности
            for j in range(series_length):
                product *= int(number_string[i + j])
            # Если текущее произведение больше `max_product`, обновляем значение `max_product`
            if product > max_product:
                max_product = product
        # Код возвращает наибольшее найденное произведение
        return max_product
    except Exception as e:
        logger.error(f'Произошла ошибка при вычислении наибольшего произведения: {e}')
        return 0
