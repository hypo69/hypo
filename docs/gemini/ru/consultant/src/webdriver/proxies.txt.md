```python
"""
Модуль для хранения списка прокси-серверов.
=========================================================================================

Этот модуль содержит список прокси-серверов, используемых для работы с webdriver.

Пример использования
--------------------

Пример импорта списка прокси:

.. code-block:: python

    from src.webdriver.proxies import PROXIES
    print(PROXIES)
"""

from src.logger.logger import logger
from src.utils.jjson import j_loads_ns
from pathlib import Path

def get_proxies() -> list[str]:
    """
    Считывает прокси из файла и возвращает их списком.

    Returns:
        list[str]: Список прокси-серверов.

    Raises:
       FileNotFoundError: Если файл с прокси не найден
    """
    file_path = Path(__file__).parent / 'proxies.txt'
    try:
        # Чтение прокси из файла с использованием j_loads_ns
        with open(file_path, 'r') as f:
            proxies = [line.strip() for line in f]
        return proxies
    except FileNotFoundError as ex:
        # Логирование ошибки, если файл не найден
        logger.error(f"Файл прокси не найден: {file_path}", ex)
        raise

PROXIES = get_proxies()
```
