"""

Этот модуль содержит инструкцию по установке драйвера Chrome для автоматизированного тестирования веб-приложений.
"""

#  Инструкция по установке Chrome для тестирования
#  ==========================================================================================

#  Основные требования
#  ---------------------

#  Список необходимых компонентов для запуска тестирования на основе Chrome WebDriver.

# 1. Установка последней версии браузера Chrome
#    ---------------------------------------------
#    Убедитесь, что на вашей системе установлен браузер Chrome последней версии.


# 2. Установка ChromeDriver
#    -------------------------
#    Скачать необходимую версию драйвера ChromeDriver с официального сайта.
#    Сохранить скачанный файл в удобное для вас место.
#    Добавить путь к папке с драйвером в переменные окружения PATH.

# 3.  Добавление пути к Chromedriver в переменные окружения
#      --------------------------------------------------
#      Для корректной работы с WebDriver необходимо добавить путь к исполняемому файлу Chromedriver в переменные окружения PATH.


# 4.  Проверка корректной работы WebDriver
#      --------------------------------------
#      В консоли проверьте работу с помощью кода ниже.
#      Если код выполняется без ошибок, то Chromedriver установлен и настроен корректно.
#      Если возникнут ошибки, проверьте правильность установки и настроек.

#  Пример проверки работоспособности
#  ----------------------------------
#  Следующий код выполняет проверку работоспособности.
#  Он запускает браузер Chrome и выполняет простую команду.
from selenium import webdriver
import time
from src.logger import logger

# # Путь к исполняемому файлу ChromeDriver.  Измените путь на свой
# chrome_driver_path = r'C:\chromedriver.exe' #  Вставьте свой путь к драйверу

def check_chromedriver():
    """Проверяет установку и настройку ChromeDriver."""
    try:
        # Инициализация драйвера ChromeDriver
        driver = webdriver.Chrome(executable_path=r'C:\chromedriver.exe') #  Вставьте свой путь к драйверу
        # Открытие новой вкладки
        driver.get("https://www.google.com")
        # Проверка, что страница загрузилась
        time.sleep(5)
        # Закрытие браузера
        driver.quit()
        return True
    except Exception as e:
        logger.error('Ошибка при проверке ChromeDriver:', e)
        return False

if check_chromedriver():
    print("ChromeDriver успешно установлен и настроен.")
else:
    print("Ошибка при установке или настройке ChromeDriver.")
