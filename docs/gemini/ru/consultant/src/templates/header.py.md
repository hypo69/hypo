"""
Модуль для определения корневой директории проекта.
=========================================================================================

Этот модуль предоставляет функцию `set_project_root`, которая определяет корневую директорию проекта,
используя заданные маркерные файлы.

Пример использования
--------------------

.. code-block:: python

    from pathlib import Path
    from src.templates.header import set_project_root

    root_path = set_project_root()
    print(root_path)

"""

import sys
from pathlib import Path
from src.logger.logger import logger #  Импорт logger

def set_project_root(marker_files: tuple = ('__root__', '.git')) -> Path:
    """
    Определяет корневую директорию проекта, начиная с директории текущего файла,
    поиском вверх по дереву каталогов до первого каталога, содержащего любой из маркерных файлов.

    :param marker_files: Кортеж имен файлов или каталогов для идентификации корневой директории проекта.
    :type marker_files: tuple
    :return: Путь к корневой директории, если она найдена, в противном случае - директория, где расположен скрипт.
    :rtype: Path

    :raises Exception: Если произошла ошибка при определении корневой директории
    
    
    Пример:
    
    .. code-block:: python

        root_path = set_project_root(marker_files = ('__root__', '.git'))
    
    """
    __root__: Path #  Объявление переменной __root__ для хранения корневой директории.
    current_path: Path = Path(__file__).resolve().parent #  Получение абсолютного пути к директории текущего файла.
    __root__ = current_path #  Установка начального значения __root__ как текущая директория.
    try:
        for parent in [current_path] + list(current_path.parents): # Перебор текущей директории и её родительских директорий.
            if any((parent / marker).exists() for marker in marker_files): #  Проверка существования маркерных файлов в каждой директории.
                __root__ = parent # Если маркерный файл найден, установить __root__ как родительскую директорию.
                break #  Завершение цикла при нахождении корневой директории.
    except Exception as ex:
        logger.error(f'Ошибка при определении корневой директории: {ex}') # Логирование ошибки, если не удалось найти корневую директорию.
        ...
    if __root__ not in sys.path: #  Проверка, не находится ли уже корневая директория в списке путей поиска модулей.
        sys.path.insert(0, str(__root__)) #  Добавление корневой директории в начало списка путей поиска модулей, чтобы обеспечить доступ к модулям проекта.
    return __root__ #  Возврат пути к корневой директории.

# Получение корневой директории проекта.
