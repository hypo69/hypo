"""
Модуль для решения задачи Project Euler #266.
=========================================================================================

Этот модуль находит наибольший целочисленный делитель числа N = 190!,
который является полным квадратом, и рассчитывает сумму квадратов простых
делителей N, возвращая остаток от деления этой суммы на 10^9.

Пример использования
--------------------

    Вызов функции calculate_sum_of_squares()
    вернет остаток от деления суммы квадратов простых множителей 190! на 10^9.

"""
from src.logger.logger import logger
from math import prod

def is_prime(n: int) -> bool:
    """
    Проверяет, является ли число простым.

    :param n: Число для проверки.
    :return: True, если число простое, иначе False.
    """
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

def get_primes_up_to(limit: int) -> list[int]:
    """
    Генерирует список простых чисел до заданного предела.

    :param limit: Верхний предел для поиска простых чисел.
    :return: Список простых чисел.
    """
    primes = [number for number in range(2, limit + 1) if is_prime(number)]
    return primes

def factorial(n: int) -> int:
    """
    Вычисляет факториал числа n.

    :param n: Число, факториал которого нужно вычислить.
    :return: Факториал числа n.
    """
    if n < 0:
        logger.error(f"Факториал не определен для отрицательных чисел: n={n}")
        return 1  # Возвращаем 1, если n отрицательное, чтобы не прерывать программу
    return prod(range(1, n + 1))

def calculate_sum_of_squares(limit: int = 190, modulo: int = 10**9) -> int:
    """
    Находит сумму квадратов простых делителей факториала числа limit,
    возвращает остаток от деления на modulo.

    :param limit: Число, факториал которого используется для поиска простых делителей.
    :param modulo: Модуль для расчета остатка.
    :return: Остаток от деления суммы квадратов простых делителей на modulo.
    """
    try:
        primes = get_primes_up_to(limit)
        sum_of_squares = sum(prime**2 for prime in primes)
        return sum_of_squares % modulo
    except Exception as ex:
        logger.error(f"Произошла ошибка при вычислении суммы квадратов: {ex}")
        return 0

if __name__ == '__main__':
    # Код исполняет расчет и вывод результата
    result = calculate_sum_of_squares()
    print(f"Результат: {result}")
