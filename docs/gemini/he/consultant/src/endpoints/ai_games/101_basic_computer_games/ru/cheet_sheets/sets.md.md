```python
"""

—Ç–∞–∫–∏–µ –∫–∞–∫ –æ–±—ä–µ–¥–∏–Ω–µ–Ω–∏–µ, –ø–µ—Ä–µ—Å–µ—á–µ–Ω–∏–µ, —Ä–∞–∑–Ω–æ—Å—Ç—å, —Å–∏–º–º–µ—Ç—Ä–∏—á–µ—Å–∫–∞—è —Ä–∞–∑–Ω–æ—Å—Ç—å,
–ø—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –ø–æ–¥–º–Ω–æ–∂–µ—Å—Ç–≤–æ –∏ –Ω–∞–¥–º–Ω–æ–∂–µ—Å—Ç–≤–æ, –∞ —Ç–∞–∫–∂–µ –ø—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –Ω–∞–ª–∏—á–∏–µ —ç–ª–µ–º–µ–Ω—Ç–∞.

from typing import Set
# from src.utils.jjson import j_loads, j_loads_ns # –ü—Ä–µ–¥–ø–æ–ª–∞–≥–∞–µ—Ç—Å—è –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ, –Ω–æ –≤ —Ç–µ–∫—É—â–µ–º –∫–æ–¥–µ –Ω–µ –Ω—É–∂–Ω–æ.
from src.logger.logger import logger # –î–æ–±–∞–≤–ª–µ–Ω –∏–º–ø–æ—Ä—Ç –ª–æ–≥–≥–µ—Ä–∞

def create_fruit_set(fruit_string: str) -> Set[str]:
    """
    –°–æ–∑–¥–∞–µ—Ç –º–Ω–æ–∂–µ—Å—Ç–≤–æ —Ñ—Ä—É–∫—Ç–æ–≤ –∏–∑ —Å—Ç—Ä–æ–∫–∏.

    :param fruit_string: –°—Ç—Ä–æ–∫–∞ —Å —Ñ—Ä—É–∫—Ç–∞–º–∏ (üçé, üçê, üçâ, üß∫).
    :type fruit_string: str
    :raises ValueError: –ï—Å–ª–∏ —Å—Ç—Ä–æ–∫–∞ —Å–æ–¥–µ—Ä–∂–∏—Ç –Ω–µ–¥–æ–ø—É—Å—Ç–∏–º—ã–µ —Å–∏–º–≤–æ–ª—ã.
    :return: –ú–Ω–æ–∂–µ—Å—Ç–≤–æ —É–Ω–∏–∫–∞–ª—å–Ω—ã—Ö —Ñ—Ä—É–∫—Ç–æ–≤.
    :rtype: Set[str]

    –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è:
    
    .. code-block:: python

      create_fruit_set("üçéüçêüçâ")  # –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç: {"üçé", "üçê", "üçâ"}
    """
    # –ü—Ä–æ–≤–µ—Ä–∫–∞, —á—Ç–æ —Å—Ç—Ä–æ–∫–∞ —Å–æ–¥–µ—Ä–∂–∏—Ç —Ç–æ–ª—å–∫–æ –¥–æ–ø—É—Å—Ç–∏–º—ã–µ —Å–∏–º–≤–æ–ª—ã
    if not all(fruit in ["üçé", "üçê", "üçâ", "üß∫"] for fruit in fruit_string): # –ø—Ä–æ–≤–µ—Ä—è–µ–º —á—Ç–æ —Å—Ç—Ä–æ–∫–∞ —Å–æ—Å—Ç–æ–∏—Ç —Ç–æ–ª—å–∫–æ –∏–∑ —Ä–∞–∑—Ä–µ—à–µ–Ω–Ω—ã—Ö —Å–∏–º–≤–æ–ª–æ–≤ —é–Ω–∏–∫–æ–¥–∞
        logger.error(f"–°—Ç—Ä–æ–∫–∞ –º–æ–∂–µ—Ç —Å–æ–¥–µ—Ä–∂–∞—Ç—å —Ç–æ–ª—å–∫–æ —Å–∏–º–≤–æ–ª—ã üçé, üçê, üçâ, üß∫. –ü–æ–ª—É—á–µ–Ω–æ: {fruit_string}") # –ò—Å–ø–æ–ª—å–∑—É–µ–º logger –¥–ª—è –ª–æ–≥–∏—Ä–æ–≤–∞–Ω–∏—è –æ—à–∏–±–∫–∏
        raise ValueError("–°—Ç—Ä–æ–∫–∞ –º–æ–∂–µ—Ç —Å–æ–¥–µ—Ä–∂–∞—Ç—å —Ç–æ–ª—å–∫–æ —Å–∏–º–≤–æ–ª—ã üçé, üçê, üçâ, üß∫") # –í—ã–±—Ä–∞—Å—ã–≤–∞–µ–º –∏—Å–∫–ª—é—á–µ–Ω–∏–µ –µ—Å–ª–∏ —Å—Ç—Ä–æ–∫–∞ –Ω–µ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É–µ—Ç —Ç—Ä–µ–±–æ–≤–∞–Ω–∏—è–º
    return set(fruit_string) # –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º —Å—Ç—Ä–æ–∫—É –≤ –º–Ω–æ–∂–µ—Å—Ç–≤–æ –∏ –≤–æ–∑–≤—Ä–∞—â–∞–µ–º –µ–≥–æ

def display_set(fruit_set: Set[str]) -> str:
    """
    –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ—Ç –º–Ω–æ–∂–µ—Å—Ç–≤–æ —Ñ—Ä—É–∫—Ç–æ–≤ –≤ —Å—Ç—Ä–æ–∫—É –¥–ª—è –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏—è.

    :param fruit_set: –ú–Ω–æ–∂–µ—Å—Ç–≤–æ —Ñ—Ä—É–∫—Ç–æ–≤.
    :type fruit_set: Set[str]
    :return: –°—Ç—Ä–æ–∫–∞ –¥–ª—è –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏—è –º–Ω–æ–∂–µ—Å—Ç–≤–∞.
    :rtype: str

    –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è:
    
    .. code-block:: python

    """
    # –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º –º–Ω–æ–∂–µ—Å—Ç–≤–æ –≤ —Å—Ç—Ä–æ–∫—É –¥–ª—è –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏—è
    return "{" + ", ".join(fruit_set) + "}" # —Ñ–æ—Ä–º–∏—Ä—É–µ–º —Å—Ç—Ä–æ–∫—É –¥–ª—è –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏—è –º–Ω–æ–∂–µ—Å—Ç–≤–∞

# –°–æ–∑–¥–∞–µ–º –º–Ω–æ–∂–µ—Å—Ç–≤–∞ —Ñ—Ä—É–∫—Ç–æ–≤
fruits_set_A = create_fruit_set("üçéüçê") # –ú–Ω–æ–∂–µ—Å—Ç–≤–æ A: {üçé, üçê}
fruits_set_B = create_fruit_set("üçêüçâ") # –ú–Ω–æ–∂–µ—Å—Ç–≤–æ B: {üçê, üçâ}
fruits_set_C = create_fruit_set("üçéüçêüçâ") # –ú–Ω–æ–∂–µ—Å—Ç–≤–æ C: {üçé, üçê, üçâ}
fruits_set_D = create_fruit_set("üß∫") # –ú–Ω–æ–∂–µ—Å—Ç–≤–æ D: {üß∫}

# –í—ã–≤–æ–¥–∏–º –º–Ω–æ–∂–µ—Å—Ç–≤–∞
print(f"–ú–Ω–æ–∂–µ—Å—Ç–≤–æ A: {display_set(fruits_set_A)}") # –í—ã–≤–æ–¥–∏–º –º–Ω–æ–∂–µ—Å—Ç–≤–æ A
print(f"–ú–Ω–æ–∂–µ—Å—Ç–≤–æ B: {display_set(fruits_set_B)}") # –í—ã–≤–æ–¥–∏–º –º–Ω–æ–∂–µ—Å—Ç–≤–æ B
print(f"–ú–Ω–æ–∂–µ—Å—Ç–≤–æ C: {display_set(fruits_set_C)}") # –í—ã–≤–æ–¥–∏–º –º–Ω–æ–∂–µ—Å—Ç–≤–æ C
print(f"–ú–Ω–æ–∂–µ—Å—Ç–≤–æ D: {display_set(fruits_set_D)}") # –í—ã–≤–æ–¥–∏–º –º–Ω–æ–∂–µ—Å—Ç–≤–æ D

# –û–±—ä–µ–¥–∏–Ω–µ–Ω–∏–µ –º–Ω–æ–∂–µ—Å—Ç–≤ (Union)
union_result = fruits_set_A | fruits_set_B # –≤—ã–ø–æ–ª–Ω—è–µ–º –æ–ø–µ—Ä–∞—Ü–∏—é –æ–±—ä–µ–¥–∏–Ω–µ–Ω–∏—è –º–Ω–æ–∂–µ—Å—Ç–≤
print(f"A ‚à™ B: {display_set(union_result)}")  # –†–µ–∑—É–ª—å—Ç–∞—Ç: {üçé, üçê, üçâ} # –í—ã–≤–æ–¥–∏–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç –æ–±—ä–µ–¥–∏–Ω–µ–Ω–∏—è

# –ü–µ—Ä–µ—Å–µ—á–µ–Ω–∏–µ –º–Ω–æ–∂–µ—Å—Ç–≤ (Intersection)
intersection_result = fruits_set_A & fruits_set_B # –≤—ã–ø–æ–ª–Ω—è–µ–º –æ–ø–µ—Ä–∞—Ü–∏—é –ø–µ—Ä–µ—Å–µ—á–µ–Ω–∏—è –º–Ω–æ–∂–µ—Å—Ç–≤
print(f"A ‚à© B: {display_set(intersection_result)}")  # –†–µ–∑—É–ª—å—Ç–∞—Ç: {üçê} # –í—ã–≤–æ–¥–∏–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç –ø–µ—Ä–µ—Å–µ—á–µ–Ω–∏—è

# –†–∞–∑–Ω–æ—Å—Ç—å –º–Ω–æ–∂–µ—Å—Ç–≤ (Difference)
difference_result_AB = fruits_set_A - fruits_set_B # –≤—ã–ø–æ–ª–Ω—è–µ–º –æ–ø–µ—Ä–∞—Ü–∏—é —Ä–∞–∑–Ω–æ—Å—Ç–∏ –º–Ω–æ–∂–µ—Å—Ç–≤ A –∏ B
print(f"A - B: {display_set(difference_result_AB)}")  # –†–µ–∑—É–ª—å—Ç–∞—Ç: {üçé} # –í—ã–≤–æ–¥–∏–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç —Ä–∞–∑–Ω–æ—Å—Ç–∏ A –∏ B
difference_result_BA = fruits_set_B - fruits_set_A # –≤—ã–ø–æ–ª–Ω—è–µ–º –æ–ø–µ—Ä–∞—Ü–∏—é —Ä–∞–∑–Ω–æ—Å—Ç–∏ –º–Ω–æ–∂–µ—Å—Ç–≤ B –∏ A
print(f"B - A: {display_set(difference_result_BA)}")  # –†–µ–∑—É–ª—å—Ç–∞—Ç: {üçâ} # –í—ã–≤–æ–¥–∏–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç —Ä–∞–∑–Ω–æ—Å—Ç–∏ B –∏ A

# –°–∏–º–º–µ—Ç—Ä–∏—á–Ω–∞—è —Ä–∞–∑–Ω–æ—Å—Ç—å –º–Ω–æ–∂–µ—Å—Ç–≤ (Symmetric Difference)
symmetric_difference_result = fruits_set_A ^ fruits_set_B # –≤—ã–ø–æ–ª–Ω—è–µ–º –æ–ø–µ—Ä–∞—Ü–∏—é —Å–∏–º–º–µ—Ç—Ä–∏—á–Ω–æ–π —Ä–∞–∑–Ω–æ—Å—Ç–∏
print(f"A ^ B: {display_set(symmetric_difference_result)}")  # –†–µ–∑—É–ª—å—Ç–∞—Ç: {üçé, üçâ} # –í—ã–≤–æ–¥–∏–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç —Å–∏–º–º–µ—Ç—Ä–∏—á–Ω–æ–π —Ä–∞–∑–Ω–æ—Å—Ç–∏

# –ü–æ–¥–º–Ω–æ–∂–µ—Å—Ç–≤–æ (Subset)
subset_result1 = fruits_set_A <= fruits_set_C # –ø—Ä–æ–≤–µ—Ä—è–µ–º —è–≤–ª—è–µ—Ç—Å—è –ª–∏ A –ø–æ–¥–º–Ω–æ–∂–µ—Å—Ç–≤–æ–º C
```
