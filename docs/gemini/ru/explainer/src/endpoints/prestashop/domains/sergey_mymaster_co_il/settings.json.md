**Пример использования в коде:**

Представим, что в проекте есть функция, которая должна получить ID языка по его коду:

```python
import json

def get_lang_id(lang_code, settings_path):
    """
    Получает ID языка по его коду из файла настроек.
    
    Args:
        lang_code (str): Код языка (например, "en", "he", "ru").
        settings_path (str): Путь к файлу settings.json.
        
    Returns:
        int: ID языка или None, если язык не найден.
    """
    try:
        with open(settings_path, 'r') as f:
            settings = json.load(f)
            if 'langs' in settings and lang_code in settings['langs']:
                return settings['langs'][lang_code]
            else:
              return None
    except FileNotFoundError:
      return None

settings_file = "hypotez/src/endpoints/prestashop/domains/sergey_mymaster_co_il/settings.json"
english_id = get_lang_id("en",settings_file) # english_id будет равен 1
russian_id = get_lang_id("ru",settings_file)  # russian_id будет равен 3
not_found_id = get_lang_id("fr",settings_file) # not_found_id будет равен None

print(f"English ID: {english_id}, Russian ID: {russian_id}, French ID: {not_found_id}")


```
