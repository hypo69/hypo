```python
def dec_to_hex(decimal: int) -> str:
    """
    –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ—Ç –¥–µ—Å—è—Ç–∏—á–Ω–æ–µ —á–∏—Å–ª–æ (—Ü–µ–ª–æ–µ) –≤ —à–µ—Å—Ç–Ω–∞–¥—Ü–∞—Ç–µ—Ä–∏—á–Ω–æ–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–∏–µ (—Å—Ç—Ä–æ–∫—É).

    """
    if decimal == 0:

    hex_digits = "0123456789ABCDEF"
    hexadecimal = ""

    return hexadecimal
```
