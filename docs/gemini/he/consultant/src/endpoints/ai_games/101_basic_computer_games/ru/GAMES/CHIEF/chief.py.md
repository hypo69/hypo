"""
Модуль CHIEF:
=================

Модуль представляет собой игру "CHIEF", в которой игрок угадывает три числа, сгенерированные компьютером.

Правила игры:
1. Компьютер генерирует три случайных числа от 1 до 10.
2. Игрок вводит свои предположения для этих чисел.
3. Компьютер сообщает, какие из введенных чисел не совпадают с загаданными.
4. Игра продолжается, пока игрок не угадает все три числа.

Пример использования:

.. code-block:: python

    import random
    from src.logger.logger import logger

    # ... (основной код игры)
"""
import random # Импортирует модуль random для генерации случайных чисел
from src.logger.logger import logger # Импортирует logger для записи ошибок

# Генерирует случайные числа от 1 до 10
targetA = random.randint(1, 10) # Загаданное число A
targetB = random.randint(1, 10) # Загаданное число B
targetC = random.randint(1, 10) # Загаданное число C

while True: # Бесконечный цикл, пока игрок не угадает все числа
    try:
        # Запрашивает ввод трех чисел от пользователя
        userA = int(input('Введите число для A (от 1 до 10): ')) # Ввод числа A от пользователя
        userB = int(input('Введите число для B (от 1 до 10): ')) # Ввод числа B от пользователя
        userC = int(input('Введите число для C (от 1 до 10): ')) # Ввод числа C от пользователя
    except ValueError as ex:
        # Обрабатывает ошибку, если ввод не является целым числом
        logger.error('Пожалуйста, введите целые числа.', ex) # Сообщение об ошибке
        continue

    message = '' # Инициализирует строку для хранения информации об ошибках

    # Проверяет, верно ли число A
    if userA != targetA: # Проверка числа A
        message += 'A' # Добавляет "A" в сообщение об ошибке

    # Проверяет, верно ли число B
    if userB != targetB: # Проверка числа B
        message += 'B' # Добавляет "B" в сообщение об ошибке

    # Проверяет, верно ли число C
    if userC != targetC: # Проверка числа C
        message += 'C' # Добавляет "C" в сообщение об ошибке

    # Выводит сообщение об ошибках или поздравление
    if message: # Проверяет, есть ли ошибки
        print(f'WRONG ON {message}') # Вывод сообщения об ошибке
    else:
        print('YOU GOT IT!') # Вывод сообщения об успехе
        break # Завершает игру
