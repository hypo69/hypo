"""
Модуль для решения задачи 790 Project Euler "Нечётные делители"
=========================================================================================

Определяет функции для вычисления количества нечетных делителей числа,
суммы количества нечетных делителей до заданного числа,
и вычисления чисел Фибоначчи.
"""
from src.logger.logger import logger
from functools import lru_cache

def count_odd_divisors(n: int) -> int:
    """
    Вычисляет количество нечетных делителей числа n.

    :param n: Целое число, для которого вычисляются нечетные делители.
    :return: Количество нечетных делителей числа n.
    """
    if n < 1:
        logger.error(f"Некорректный ввод: n = {n}. Число должно быть положительным.")
        return 0
    count = 0
    for i in range(1, int(n**0.5) + 1, 2):
        if n % i == 0:
            count += 1
            if i * i != n and (n // i) % 2 != 0:
                count += 1
    return count

def sum_odd_divisors(n: int) -> int:
    """
    Вычисляет сумму количества нечетных делителей для всех чисел от 1 до n.

    :param n: Верхняя граница диапазона чисел.
    :return: Сумма количества нечетных делителей.
    """
    if n < 1:
        logger.error(f"Некорректный ввод: n = {n}. Число должно быть положительным.")
        return 0
    total_sum = 0
    for i in range(1, n + 1):
        total_sum += count_odd_divisors(i)
    return total_sum

@lru_cache(maxsize=None)
def fibonacci(n: int) -> int:
    """
    Вычисляет n-е число Фибоначчи.

    :param n: Порядковый номер числа Фибоначчи (начиная с 1).
    :return: n-е число Фибоначчи.
    """
    if n <= 0:
        logger.error(f"Некорректный ввод: n = {n}. Число должно быть положительным.")
        return 0
    if n <= 2:
        return 1
    return (fibonacci(n - 1) + fibonacci(n - 2))

def solve_problem(n: int) -> int:
    """
    Решает задачу 790 Project Euler для заданного n, вычисляя S(F(n)) % 10^9.

    :param n: Порядковый номер числа Фибоначчи.
    :return: Результат вычислений S(F(n)) % 10^9.
    """
    if n <= 0:
        logger.error(f"Некорректный ввод: n = {n}. Число должно быть положительным.")
        return 0
    try:
        fib_n = fibonacci(n)
        result = sum_odd_divisors(fib_n) % 1000000000
        return result
    except Exception as ex:
        logger.error(f"Ошибка при решении задачи: {ex}")
        return 0

if __name__ == '__main__':
    n_value = 10**3  # Используем меньшее значение для теста
    result = solve_problem(n_value)
    print(f"S(F({n_value})) % 10^9 = {result}")
