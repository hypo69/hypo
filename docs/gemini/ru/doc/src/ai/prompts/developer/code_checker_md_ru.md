def my_function(param1: str, param2: int = 0) -> str:
    """
    Функция выполняет умножение строки на целое число.

    :param param1: Строка, которая будет умножаться.
    :type param1: str
    :param param2: Целое число, на которое умножается строка. По умолчанию 0.
    :type param2: int
    :returns: Строковое представление результата умножения. Возвращает пустую строку, если param2 равно 0 или если param1 не является строкой.
    :rtype: str
    """
    if not isinstance(param1, str):
        return ""
    if param2 == 0:
        return ""
    result = param1 * param2
    return str(result)
```
