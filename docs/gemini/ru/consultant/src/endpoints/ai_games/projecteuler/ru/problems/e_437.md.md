"""
Модуль для решения задачи 437 проекта Эйлера.
=================================================

Этот модуль содержит функции для вычисления чисел Фибоначчи,
последовательностей G(a, b, n), определения принадлежности простых чисел
гиперболической группе Фибоначчи и нахождения n-го простого числа,
не являющегося членом этой группы.

Пример использования
--------------------

.. code-block:: python

    from src.endpoints.ai_games.projecteuler.ru.problems.e_437 import find_h_n
    result = find_h_n(100000)
    print(f'H(100000) = {result}')
"""
from typing import List
from math import sqrt
from src.logger.logger import logger

def fibonacci(n: int) -> int:
    """
    Вычисляет n-е число Фибоначчи.

    :param n: Порядковый номер числа Фибоначчи (начиная с 0).
    :return: n-е число Фибоначчи.
    """
    if n <= 1:
        return n
    a, b = 0, 1
    for _ in range(2, n + 1):
        a, b = b, a + b
    return b

def g_sequence(a: int, b: int, n: int) -> int:
    """
    Вычисляет n-й член последовательности G(a, b, n).

    :param a: Первый элемент последовательности.
    :param b: Второй элемент последовательности.
    :param n: Порядковый номер элемента последовательности (начиная с 0).
    :return: n-й член последовательности G(a, b, n).
    """
    if n == 0:
        return a
    if n == 1:
        return b
    prev_prev, prev = a, b
    for _ in range(2, n + 1):
        current = prev + prev_prev
        prev_prev, prev = prev, current
    return prev

def is_hyperbolic_fibonacci(p: int) -> bool:
    """
    Проверяет, является ли простое число членом гиперболической группы Фибоначчи.

    :param p: Простое число для проверки.
    :return: True, если простое число является членом группы, иначе False.
    """
    for a in range(1, p):
        for b in range(1, p):
            for k in range(1, p + 1):
                if g_sequence(a, b, k) % p == 0:
                    return True
    return False

def is_prime(n: int) -> bool:
    """
    Проверяет, является ли число простым.

    :param n: Число для проверки.
    :return: True, если число простое, иначе False.
    """
    if n <= 1:
        return False
    if n <= 3:
        return True
    if n % 2 == 0 or n % 3 == 0:
        return False
    for i in range(5, int(sqrt(n)) + 1, 6):
        if n % i == 0 or n % (i + 2) == 0:
            return False
    return True


def find_h_n(n: int) -> int:
    """
    Находит n-е простое число, не являющееся членом гиперболической группы Фибоначчи.

    :param n: Порядковый номер искомого простого числа.
    :return: n-е простое число, не входящее в группу.
    """
    count = 0
    num = 2
    while count < n:
        if is_prime(num):
            if not is_hyperbolic_fibonacci(num):
                count += 1
                if count == n:
                  return num
        num += 1
    return -1
    

if __name__ == '__main__':
    try:
        # код исполняет поиск H(100000)
        result = find_h_n(100000)
        print(f'H(100000) = {result}')
    except Exception as e:
         # Логирует ошибку, если что-то пошло не так
        logger.error(f'Произошла ошибка при вычислении H(100000): {e}')
