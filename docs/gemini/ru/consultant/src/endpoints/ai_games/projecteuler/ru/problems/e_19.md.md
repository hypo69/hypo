"""
Модуль для решения задачи Project Euler #19: Считая воскресенья.
==================================================================

Этот модуль содержит функцию для подсчета количества воскресений,
выпадающих на первое число месяца в двадцатом веке (с 1 января 1901
года по 31 декабря 2000 года).

Функция :func:`count_sundays` вычисляет количество воскресений,
выпадающих на первое число месяца.

Пример использования
--------------------

>>> count_sundays()
171
"""

from src.logger.logger import logger  # Импортируем logger
from datetime import date, timedelta # импортируем date и timedelta для работы с датами

def is_leap_year(year: int) -> bool:
    """
    Определяет, является ли год високосным.

    :param year: Год для проверки.
    :return: True, если год високосный, иначе False.
    """
    # Код выполняет проверку, является ли год високосным
    if year % 4 == 0:
        if year % 100 == 0:
            if year % 400 == 0:
                return True
            else:
                return False
        else:
            return True
    else:
        return False

def days_in_month(month: int, year: int) -> int:
    """
    Определяет количество дней в месяце заданного года.

    :param month: Номер месяца (1 - январь, 12 - декабрь).
    :param year: Год.
    :return: Количество дней в месяце.
    """
    # Код возвращает количество дней в месяце
    if month in [4, 6, 9, 11]:
        return 30
    elif month == 2:
        return 29 if is_leap_year(year) else 28
    else:
        return 31

def count_sundays() -> int:
    """
    Подсчитывает количество воскресений, выпадающих на первое число месяца
    в период с 1 января 1901 года по 31 декабря 2000 года.

    :return: Количество воскресений.
    """
    # Код инициализирует счетчик воскресений и устанавливает начальную дату
    count = 0
    current_date = date(1901, 1, 1)
    end_date = date(2000, 12, 31)

    # Цикл проходит через все даты в заданном диапазоне
    while current_date <= end_date:
        # Код проверяет, является ли первое число месяца воскресеньем
        if current_date.weekday() == 6 and current_date.day == 1:
           count += 1
        # Код вычисляет количество дней в текущем месяце и увеличивает дату на один месяц
        days = days_in_month(current_date.month, current_date.year)
        current_date += timedelta(days=days)
        # Код перемещает дату на начало следующего месяца
        current_date = date(current_date.year, current_date.month, 1)
    # Код возвращает общее количество воскресений, выпадающих на первое число месяца
    return count

if __name__ == '__main__':
    try:
        # Код исполняет функцию count_sundays и выводит результат
        result = count_sundays()
        print(f'Количество воскресений: {result}')
    except Exception as e:
        # Код отлавливает и логирует ошибку, если что-то пошло не так
        logger.error(f'Произошла ошибка: {e}')
