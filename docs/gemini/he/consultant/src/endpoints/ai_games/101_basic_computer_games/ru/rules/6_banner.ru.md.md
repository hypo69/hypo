    def create_banner(self, message: str) -> str:
        """
        Создает баннер из введенной строки.

        :param message: Строка, которую нужно преобразовать в баннер.
        :return: Баннер в виде строки.
        """
        message = message.upper()
        banner_lines = ['' for _ in range(5)] # Создаем пустые строки для вывода баннера
        for char in message:
            if char in self.chars:
                char_matrix = self.chars[char] # Получаем матрицу символов для текущего символа
                for i, line in enumerate(char_matrix):
                    banner_lines[i] += line + '  ' # Добавляем строку из матрицы символов в строку баннера
            else:
                logger.error(f'Символ {char} не поддерживается') # Логируем ошибку, если символ не найден
                continue
        return '\n'.join(banner_lines) # Возвращаем баннер в виде строки

    def run_game(self):
        """
        Запускает игру "Banner".
        """
        while True: # Запускаем бесконечный цикл игры
            message = input('Введите сообщение для создания баннера: ') # Получаем сообщение от пользователя
            if not message:
                logger.error("Сообщение не может быть пустым") # Логируем ошибку
                continue

            # Проверка, что ввод состоит только из букв и цифр
            if not all(char.isalnum() for char in message):
                logger.error("Сообщение должно содержать только буквы и цифры") # Логируем ошибку
                continue

            banner = self.create_banner(message) # Создаем баннер из введенного сообщения
            print(banner) # Выводим баннер
            another = input('Хотите создать еще один баннер? (да/нет): ') # Получаем запрос на повторную игру

            if another.lower() != 'да':
                print('Спасибо за игру!')
                break # Завершаем игру
