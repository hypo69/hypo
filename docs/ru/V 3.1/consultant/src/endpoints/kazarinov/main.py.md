"""
Модуль для запуска мини-бота Kazarinov.
========================================

Модуль содержит точку входа для запуска основной функции мини-бота.

Пример использования
----------------------

>>> python src/endpoints/kazarinov/main.py
"""
import asyncio
from src.logger import logger  # Import logger from src.logger
from src.endpoints.kazarinov.minibot import main

async def async_main():
    """
    Асинхронная функция для запуска main из minibot.
    """
    try:
        await main()  # Запускаем основную функцию minibot
    except Exception as ex:
        logger.error('Ошибка при выполнении main() в minibot', ex, exc_info=True)  # Логируем ошибку с traceback


if __name__ == "__main__":
    try:
        asyncio.run(async_main())  # Запускаем асинхронную функцию
    except Exception as ex:
        logger.error('Ошибка при запуске asyncio', ex, exc_info=True)  # Логируем ошибку с traceback
