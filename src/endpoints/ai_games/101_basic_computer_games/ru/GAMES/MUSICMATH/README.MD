*   **`find_nearest_notes(frequency, concert_a_freq=440.0)`**: –ù–∞—Ö–æ–¥–∏—Ç –¥–≤–µ –±–ª–∏–∂–∞–π—à–∏–µ –Ω–æ—Ç—ã (—Å–≤–µ—Ä—Ö—É –∏ —Å–Ω–∏–∑—É) –∫ –∑–∞–¥–∞–Ω–Ω–æ–π —á–∞—Å—Ç–æ—Ç–µ.

```python
# @title –ú–æ–¥—É–ª—å MusicMath: –ø—Ä–∏–º–µ—Ä—ã —Å–≤—è–∑–∏ –º–∞—Ç–µ–º–∞—Ç–∏–∫–∏ –∏ –º—É–∑—ã–∫–∏.

"""
–ú–æ–¥—É–ª—å MusicMath: –ø—Ä–∏–º–µ—Ä—ã —Å–≤—è–∑–∏ –º–∞—Ç–µ–º–∞—Ç–∏–∫–∏ –∏ –º—É–∑—ã–∫–∏.
"""
import math
from fractions import Fraction
import re

def calculate_frequency(note_number, concert_a_freq=440.0):
    """
    –í—ã—á–∏—Å–ª—è–µ—Ç —á–∞—Å—Ç–æ—Ç—É –Ω–æ—Ç—ã, –∑–Ω–∞—è –µ—ë –Ω–æ–º–µ—Ä –≤ —Ö—Ä–æ–º–∞—Ç–∏—á–µ—Å–∫–æ–π –≥–∞–º–º–µ.
    
    Args:
        note_number (int): –ù–æ–º–µ—Ä –Ω–æ—Ç—ã –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ A4 (A4 = 0).
        concert_a_freq (float, optional): –ß–∞—Å—Ç–æ—Ç–∞ –Ω–æ—Ç—ã A4. Defaults to 440.0.
    
    Returns:
        float: –ß–∞—Å—Ç–æ—Ç–∞ –Ω–æ—Ç—ã –≤ –≥–µ—Ä—Ü–∞—Ö.
    """
    return concert_a_freq * math.pow(2, note_number / 12)

def calculate_interval_ratio(note1_number, note2_number):
    """
    –í—ã—á–∏—Å–ª—è–µ—Ç –æ—Ç–Ω–æ—à–µ–Ω–∏–µ —á–∞—Å—Ç–æ—Ç –º–µ–∂–¥—É –¥–≤—É–º—è –Ω–æ—Ç–∞–º–∏, –∑–Ω–∞—è –∏—Ö –Ω–æ–º–µ—Ä–∞.
    
    Args:
        note1_number (int): –ù–æ–º–µ—Ä –ø–µ—Ä–≤–æ–π –Ω–æ—Ç—ã.
        note2_number (int): –ù–æ–º–µ—Ä –≤—Ç–æ—Ä–æ–π –Ω–æ—Ç—ã.
    
    Returns:
         str: –û—Ç–Ω–æ—à–µ–Ω–∏–µ —á–∞—Å—Ç–æ—Ç –≤ –≤–∏–¥–µ –ø—Ä–æ—Å—Ç–æ–π –¥—Ä–æ–±–∏.
    """
    freq1 = calculate_frequency(note1_number)
    freq2 = calculate_frequency(note2_number)
    ratio = freq2/freq1 if freq2>freq1 else freq1/freq2
    fraction_ratio = Fraction(ratio).limit_denominator(100)  # Limit to reasonable denominators
    return str(fraction_ratio)

def calculate_tempo_duration(bpm, beat_length, beats):
     """
    –í—ã—á–∏—Å–ª—è–µ—Ç –æ–±—â—É—é –ø—Ä–æ–¥–æ–ª–∂–∏—Ç–µ–ª—å–Ω–æ—Å—Ç—å –≤ —Å–µ–∫—É–Ω–¥–∞—Ö –º—É–∑—ã–∫–∞–ª—å–Ω–æ–≥–æ —Ñ—Ä–∞–≥–º–µ–Ω—Ç–∞.

    Args:
         bpm (int): –¢–µ–º–ø –≤ —É–¥–∞—Ä–∞—Ö –≤ –º–∏–Ω—É—Ç—É.
         beat_length(float): –ü—Ä–æ–¥–æ–ª–∂–∏—Ç–µ–ª—å–Ω–æ—Å—Ç—å –æ–¥–Ω–æ–≥–æ —É–¥–∞—Ä–∞ –≤ —Å–µ–∫—É–Ω–¥–∞—Ö.
         beats (int): –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ —É–¥–∞—Ä–æ–≤ –≤ –º—É–∑—ã–∫–∞–ª—å–Ω–æ–º —Ñ—Ä–∞–≥–º–µ–Ω—Ç–µ.

    Returns:
        float: –û–±—â–∞—è –ø—Ä–æ–¥–æ–ª–∂–∏—Ç–µ–ª—å–Ω–æ—Å—Ç—å —Ñ—Ä–∞–≥–º–µ–Ω—Ç–∞ –≤ —Å–µ–∫—É–Ω–¥–∞—Ö.
    """
     return beat_length*beats

def calculate_note_duration(tempo, note_value):
    """
    –í—ã—á–∏—Å–ª—è–µ—Ç –¥–ª–∏—Ç–µ–ª—å–Ω–æ—Å—Ç—å –Ω–æ—Ç—ã –≤ —Å–µ–∫—É–Ω–¥–∞—Ö.
    
    Args:
        tempo (int): –¢–µ–º–ø –≤ —É–¥–∞—Ä–∞—Ö –≤ –º–∏–Ω—É—Ç—É.
        note_value (float): –î–ª–∏—Ç–µ–ª—å–Ω–æ—Å—Ç—å –Ω–æ—Ç—ã –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ —Ü–µ–ª–æ–π –Ω–æ—Ç—ã (1.0 = —Ü–µ–ª–∞—è, 0.5 = –ø–æ–ª–æ–≤–∏–Ω–Ω–∞—è, 0.25 = —á–µ—Ç–≤–µ—Ä—Ç–Ω–∞—è –∏ —Ç.–¥.).

    Returns:
        float: –î–ª–∏—Ç–µ–ª—å–Ω–æ—Å—Ç—å –Ω–æ—Ç—ã –≤ —Å–µ–∫—É–Ω–¥–∞—Ö.
    """
    beat_duration = 60 / tempo
    return beat_duration * note_value

def calculate_rhythm_pattern(bar_length, note_values):
    """
    –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ—Ç –æ–±—â—É—é –ø—Ä–æ–¥–æ–ª–∂–∏—Ç–µ–ª—å–Ω–æ—Å—Ç—å —Ä–∏—Ç–º–∏—á–µ—Å–∫–æ–≥–æ –ø–∞—Ç—Ç–µ—Ä–Ω–∞ –≤ —Å–µ–∫—É–Ω–¥–∞—Ö.
    
    Args:
         bar_length (float): –î–ª–∏—Ç–µ–ª—å–Ω–æ—Å—Ç—å —Ç–∞–∫—Ç–∞ –≤ –∫–æ–ª–∏—á–µ—Å—Ç–≤–µ —Ü–µ–ª—ã—Ö –Ω–æ—Ç.
         note_values (list): –°–ø–∏—Å–æ–∫ –¥–ª–∏—Ç–µ–ª—å–Ω–æ—Å—Ç–µ–π –Ω–æ—Ç (–æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ —Ü–µ–ª–æ–π –Ω–æ—Ç—ã)
    
    Returns:
         float: –û–±—â–∞—è –¥–ª–∏—Ç–µ–ª—å–Ω–æ—Å—Ç—å —Ä–∏—Ç–º–∏—á–µ—Å–∫–æ–≥–æ –ø–∞—Ç—Ç–µ—Ä–Ω–∞ –≤ —Å–µ–∫—É–Ω–¥–∞—Ö.
    """
    pattern_duration = 0
    for value in note_values:
       pattern_duration += bar_length * value
    return pattern_duration

def calculate_note_number(frequency, concert_a_freq=440.0):
    """
    –í—ã—á–∏—Å–ª—è–µ—Ç –Ω–æ–º–µ—Ä –Ω–æ—Ç—ã –≤ —Ö—Ä–æ–º–∞—Ç–∏—á–µ—Å–∫–æ–π –≥–∞–º–º–µ –Ω–∞ –æ—Å–Ω–æ–≤–µ —á–∞—Å—Ç–æ—Ç—ã.
    
    Args:
        frequency (float): –ß–∞—Å—Ç–æ—Ç–∞ –Ω–æ—Ç—ã –≤ –≥–µ—Ä—Ü–∞—Ö.
        concert_a_freq (float, optional): –ß–∞—Å—Ç–æ—Ç–∞ –Ω–æ—Ç—ã A4. Defaults to 440.0.
    
    Returns:
        int: –ù–æ–º–µ—Ä –Ω–æ—Ç—ã –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ A4 (A4 = 0).
    """
    return round(12 * math.log2(frequency / concert_a_freq))

def generate_scale_frequencies(root_note_number, scale_pattern, concert_a_freq=440.0):
    """
    –ì–µ–Ω–µ—Ä–∏—Ä—É–µ—Ç —á–∞—Å—Ç–æ—Ç—ã –Ω–æ—Ç –∑–∞–¥–∞–Ω–Ω–æ–π –≥–∞–º–º—ã.
    
    Args:
        root_note_number (int): –ù–æ–º–µ—Ä –∫–æ—Ä–Ω–µ–≤–æ–π –Ω–æ—Ç—ã.
        scale_pattern (list): –®–∞–±–ª–æ–Ω –≥–∞–º–º—ã (–ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å –ø–æ–ª—É—Ç–æ–Ω–æ–≤).
        concert_a_freq (float, optional): –ß–∞—Å—Ç–æ—Ç–∞ –Ω–æ—Ç—ã A4. Defaults to 440.0.
    
    Returns:
        list: –°–ø–∏—Å–æ–∫ —á–∞—Å—Ç–æ—Ç –Ω–æ—Ç –≤ –≥–∞–º–º–µ.
    """
    frequencies = []
    current_note = root_note_number
    for interval in scale_pattern:
        frequencies.append(calculate_frequency(current_note, concert_a_freq))
        current_note += interval
    return frequencies

def calculate_tuning_deviation(target_frequency, actual_frequency):
    """
    –í—ã—á–∏—Å–ª—è–µ—Ç –æ—Ç–∫–ª–æ–Ω–µ–Ω–∏–µ —á–∞—Å—Ç–æ—Ç—ã –æ—Ç —Ü–µ–ª–µ–≤–æ–π –≤ –ø—Ä–æ—Ü–µ–Ω—Ç–∞—Ö.
    
    Args:
        target_frequency (float): –¶–µ–ª–µ–≤–∞—è —á–∞—Å—Ç–æ—Ç–∞.
        actual_frequency (float): –§–∞–∫—Ç–∏—á–µ—Å–∫–∞—è —á–∞—Å—Ç–æ—Ç–∞.

    Returns:
        float: –ü—Ä–æ—Ü–µ–Ω—Ç –æ—Ç–∫–ª–æ–Ω–µ–Ω–∏—è.
    """
    return ((actual_frequency - target_frequency) / target_frequency) * 100

def get_note_name(note_number):
    """
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –Ω–∞–∑–≤–∞–Ω–∏–µ –Ω–æ—Ç—ã –ø–æ –µ—ë –Ω–æ–º–µ—Ä—É –≤ —Ö—Ä–æ–º–∞—Ç–∏—á–µ—Å–∫–æ–π –≥–∞–º–º–µ.

    Args:
        note_number (int): –ù–æ–º–µ—Ä –Ω–æ—Ç—ã –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ A4 (A4 = 0).

    Returns:
        str: –ù–∞–∑–≤–∞–Ω–∏–µ –Ω–æ—Ç—ã (–Ω–∞–ø—Ä–∏–º–µ—Ä, "A4", "C#5", "Bb3").
    """
    notes = ["A", "A#", "B", "C", "C#", "D", "D#", "E", "F", "F#", "G", "G#"]
    octave = (note_number // 12) + 4  # A4 —è–≤–ª—è–µ—Ç—Å—è 0, –ø–æ—ç—Ç–æ–º—É –¥–æ–±–∞–≤–ª—è–µ–º 4
    note_index = note_number % 12
    return f"{notes[note_index]}{octave}"

def get_note_info_from_freq(freq):
    """
    –í—ã–≤–æ–¥–∏—Ç –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏—é –æ –Ω–æ—Ç–µ –ø–æ –µ—ë —á–∞—Å—Ç–æ—Ç–µ.

    Args:
        freq (float): –ß–∞—Å—Ç–æ—Ç–∞ –Ω–æ—Ç—ã –≤ –≥–µ—Ä—Ü–∞—Ö.
    """
    note_num = calculate_note_number(freq)
    note_name = get_note_name(note_num)
    print(f"–ß–∞—Å—Ç–æ—Ç–∞ {freq:.2f} Hz, —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É–µ—Ç –Ω–æ—Ç–µ {note_name}, –Ω–æ–º–µ—Ä {note_num}")


def tune_instrument(target_freq, actual_freq):
    """
    –í—ã–≤–æ–¥–∏—Ç –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏—é –æ —Ç–æ–º, –Ω–∞—Å–∫–æ–ª—å–∫–æ –∏–Ω—Å—Ç—Ä—É–º–µ–Ω—Ç –æ—Ç–∫–ª–æ–Ω—è–µ—Ç—Å—è –æ—Ç —Ü–µ–ª–µ–≤–æ–π —á–∞—Å—Ç–æ—Ç—ã.

    Args:
        target_freq (float): –¶–µ–ª–µ–≤–∞—è —á–∞—Å—Ç–æ—Ç–∞.
        actual_freq (float): –§–∞–∫—Ç–∏—á–µ—Å–∫–∞—è —á–∞—Å—Ç–æ—Ç–∞.
    """
    dev = calculate_tuning_deviation(target_freq, actual_freq)
    if dev > 0:
        print(f"–ó–∞–≤—ã—à–µ–Ω–∏–µ –Ω–∞ {dev:.2f}%. –ù—É–∂–Ω–æ –ø–æ–Ω–∏–∑–∏—Ç—å")
    elif dev < 0:
        print(f"–ó–∞–Ω–∏–∂–µ–Ω–∏–µ –Ω–∞ {abs(dev):.2f}%. –ù—É–∂–Ω–æ –ø–æ–≤—ã—Å–∏—Ç—å")
    else:
        print("–ò–¥–µ–∞–ª—å–Ω–æ –Ω–∞—Å—Ç—Ä–æ–µ–Ω–æ!")

def note_name_to_number(note_name):
    """
    –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ—Ç –Ω–∞–∑–≤–∞–Ω–∏–µ –Ω–æ—Ç—ã (–Ω–∞–ø—Ä–∏–º–µ—Ä, "A4") –≤ –µ—ë –Ω–æ–º–µ—Ä –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ A4 (A4 = 0).

    Args:
        note_name (str): –ù–∞–∑–≤–∞–Ω–∏–µ –Ω–æ—Ç—ã (–Ω–∞–ø—Ä–∏–º–µ—Ä, "A4", "C#5", "Bb3").

    Returns:
        int: –ù–æ–º–µ—Ä –Ω–æ—Ç—ã –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ A4 (A4 = 0), –∏–ª–∏ None, –µ—Å–ª–∏ –≤–≤–æ–¥ –Ω–µ–∫–æ—Ä—Ä–µ–∫—Ç–µ–Ω.
    """
    notes = ["A", "A#", "B", "C", "C#", "D", "D#", "E", "F", "F#", "G", "G#"]
    match = re.match(r"([A-Ga-g]#?)(-?\d+)", note_name)
    if not match:
        return None
    note = match.group(1).upper()
    octave = int(match.group(2))
    
    try:
        note_index = notes.index(note)
    except ValueError:
        return None
    
    return (octave - 4) * 12 + note_index

def find_nearest_notes(frequency, concert_a_freq=440.0):
    """
    –ù–∞—Ö–æ–¥–∏—Ç –¥–≤–µ –±–ª–∏–∂–∞–π—à–∏–µ –Ω–æ—Ç—ã (—Å–≤–µ—Ä—Ö—É –∏ —Å–Ω–∏–∑—É) –∫ –∑–∞–¥–∞–Ω–Ω–æ–π —á–∞—Å—Ç–æ—Ç–µ.

    Args:
        frequency (float): –ß–∞—Å—Ç–æ—Ç–∞ –Ω–æ—Ç—ã –≤ –≥–µ—Ä—Ü–∞—Ö.
        concert_a_freq (float, optional): –ß–∞—Å—Ç–æ—Ç–∞ –Ω–æ—Ç—ã A4. Defaults to 440.0.

    Returns:
      tuple: –∫–æ—Ä—Ç–µ–∂ (–∏–º—è –Ω–∏–∂–Ω–µ–π –Ω–æ—Ç—ã, –∏–º—è –≤–µ—Ä—Ö–Ω–µ–π –Ω–æ—Ç—ã).
    """
    note_number = 12 * math.log2(frequency / concert_a_freq)
    lower_note_num = math.floor(note_number)
    upper_note_num = math.ceil(note_number)

    lower_note_name = get_note_name(lower_note_num)
    upper_note_name = get_note_name(upper_note_num)
    return lower_note_name, upper_note_name
```
