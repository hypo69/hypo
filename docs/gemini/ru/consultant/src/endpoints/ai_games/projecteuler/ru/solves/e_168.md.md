"""
Модуль решения задачи Project Euler #168

Этот модуль содержит функции для решения задачи о поиске чисел,
которые могут быть "переставлены" (последняя цифра переносится в начало),
и новое число делится на исходное.

Используется итеративный подход для поиска и суммирования таких чисел
в диапазоне до 10 цифр.
"""

from src.logger.logger import logger #  Импортируем логгер для обработки ошибок.


def is_permutation_divisible(number: int) -> bool:
    """
    Проверяет, делится ли число, полученное перестановкой цифр, на исходное число.

    :param number: Исходное число для проверки.
    :return: True, если перестановка делится на исходное число без остатка, иначе False.
    """
    try:
        # Преобразуем число в строку
        number_str = str(number)
        # Переносим последнюю цифру в начало
        rotated_number_str = number_str[-1] + number_str[:-1]
        # Преобразуем новую строку в целое число
        rotated_number = int(rotated_number_str)
        # Проверяем, делится ли новое число на исходное
        return rotated_number % number == 0
    except Exception as ex:
        logger.error(f'Произошла ошибка при проверке делимости для числа {number}: {ex}')
        return False


def sum_of_permutation_divisible_numbers() -> int:
    """
    Вычисляет сумму всех чисел, удовлетворяющих условию "перестановки и делимости", до 10 цифр.

    :return: Сумма всех найденных чисел.
    """
    total_sum = 0 # Инициализируем сумму найденных чисел
    number = 11 # Начальное значение числа
    try:
         # Цикл выполняется, пока количество цифр в числе не превысит 10
        while len(str(number)) <= 10:
            # Проверяем, удовлетворяет ли текущее число условию
            if is_permutation_divisible(number):
                total_sum += number # Добавляем число к сумме, если условие выполнено
            number += 1 # Увеличиваем число на 1 для следующей итерации
    except Exception as ex:
        logger.error(f'Произошла ошибка в процессе поиска чисел: {ex}')
    return total_sum # Возвращаем итоговую сумму

# Вывод результата:
if __name__ == '__main__':
    result = sum_of_permutation_divisible_numbers() # Вычисляем сумму
    print(result) # Выводим результат
