"""
Модуль для решения задачи 44 проекта Эйлер.
=========================================================================================

Задача заключается в поиске пары пентагональных чисел, сумма и разность которых также являются пентагональными числами.

Пример использования
--------------------

.. code-block:: python

    from src.utils.jjson import j_loads_ns
    from src.logger.logger import logger

    result = find_min_pentagonal_diff()
    print(f'Минимальная разность: {result}')
"""
from src.utils.jjson import j_loads_ns
from src.logger.logger import logger
import math


def generate_pentagonal(n: int) -> int:
    """
    Генерирует n-ое пентагональное число.

    :param n: Порядковый номер пентагонального числа.
    :return: Значение n-го пентагонального числа.
    """
    return n * (3 * n - 1) // 2


def is_pentagonal(x: int) -> bool:
    """
    Проверяет, является ли число пентагональным.

    :param x: Число для проверки.
    :return: True, если число пентагональное, False иначе.
    """
    n = (1 + math.sqrt(1 + 24 * x)) / 6
    return n == int(n)


def find_min_pentagonal_diff() -> int:
    """
    Находит минимальную разность D между пентагональными числами Pj и Pk,
    для которых сумма и разность являются пентагональными числами.

    :return: Минимальная разность D.
    """
    j = 1
    while True:
        j += 1
        pj = generate_pentagonal(j)
        for k in range(1, j):
            pk = generate_pentagonal(k)
            sum_p = pj + pk
            diff_p = abs(pj - pk)
            if is_pentagonal(sum_p) and is_pentagonal(diff_p):
                return diff_p
                # Код возвращает разницу между пентагональными числами
            if pk > pj:
                break
                # Код прерывает цикл, если pk больше pj


if __name__ == '__main__':
    try:
        #  Код исполняет поиск минимальной разности между пентагональными числами
        result = find_min_pentagonal_diff()
        print(f'Минимальная разность: {result}')
    except Exception as e:
        # Логирование ошибки
        logger.error(f'Произошла ошибка при выполнении: {e}')
