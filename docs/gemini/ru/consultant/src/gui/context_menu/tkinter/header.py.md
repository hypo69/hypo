```python
# -*- coding: utf-8 -*-
#! venv/Scripts/python.exe
#! venv/bin/python/python3.12

"""
Модуль для определения корневой директории проекта и добавления её в sys.path.
=========================================================================================

Этот модуль инициализирует необходимые переменные для работы с путями в проекте.
Он определяет корневую директорию проекта и добавляет ее в sys.path, чтобы обеспечить
доступ к другим модулям проекта.

Пример использования
--------------------

.. code-block:: python

    from pathlib import Path
    import sys, os

    # Получаем путь к текущему файлу
    current_file_path = Path(__file__)

    # Определяем корневую директорию проекта
    root_path = current_file_path.parent.parent.parent.parent.absolute()

    # Добавляем корневую директорию в sys.path
    sys.path.append(str(root_path))
"""

import sys
import os
from pathlib import Path
from src.logger.logger import logger  #  импортируем логгер

# Определяем корневую директорию проекта
try:
    #  Попытка определить корневую директорию проекта, обрезая строку пути до "hypotez"
    __root__: Path = Path(os.getcwd())
    __root__ = Path(str(__root__)[:str(__root__).rfind('hypotez')+7])
except Exception as e:
    #  В случае ошибки, логируем ее и устанавливаем __root__ в текущую рабочую директорию
    logger.error(f'Ошибка определения корневой директории: {e}')
    __root__ = Path(os.getcwd())

# Добавляем корневую директорию проекта в sys.path
try:
    #  Код добавляет корневую директорию в sys.path, чтобы обеспечить возможность импорта модулей
    sys.path.append(str(__root__))
except Exception as e:
    #  Логируем ошибку, если не удалось добавить путь в sys.path
    logger.error(f'Ошибка добавления пути в sys.path: {e}')
```
