"""
Модуль содержит функции для стилизации UI элементов.
=====================================================

Модуль предоставляет функции для упрощения стилизации элементов пользовательского интерфейса,
например, задание фиксированных размеров для виджетов.
"""

from PyQt6 import QtWidgets
from PyQt6.QtWidgets import QWidget

# Константы для размеров элементов UI
DEFAULT_WIDTH = 100
DEFAULT_HEIGHT = 30


def set_fixed_size(widget: QWidget, width: int, height: int) -> None:
    """
    Задает фиксированный размер для виджета.

    Args:
        widget (QWidget): Виджет, для которого устанавливается размер.
        width (int): Ширина виджета.
        height (int): Высота виджета.

    Returns:
        None
    """
    widget.setFixedSize(width, height)
