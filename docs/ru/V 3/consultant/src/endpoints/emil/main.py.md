"""
Модуль, содержащий точку входа для обработки сценариев emil-design.com.
======================================================================

Модуль вызывает функцию `main` из модуля `emil_design`, которая выполняет основную логику обработки.

Пример использования:
----------------------

# (Предполагается, что функция main() выполняет полезные действия)

from src.logger import logger  # Подключаем модуль logger
from .emil_design import main


def run():
    """
    Точка входа для запуска обработки сценариев emil-design.com.

    Вызывает функцию `main` из модуля `emil_design`.
    """
    try:
        logger.info('Starting emil-design processing...') # Логируем начало обработки
        main()
        logger.info('Emil-design processing completed.') # Логируем завершение обработки
    except Exception as ex:
        logger.error('Error during emil-design processing', ex, exc_info=True) # Логируем ошибку


if __name__ == "__main__":
    run()
