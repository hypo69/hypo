"""
Модуль для работы со списками строк.
=========================================================================================

Этот модуль содержит функции для работы со списками строк,
такие как чтение и вывод.
"""

from src.utils.jjson import j_loads
from src.logger import logger

def print_list(list_data: list) -> None:
    """
    Выводит список строк в консоль.

    :param list_data: Список строк.
    :return: None
    """
    # проверка, что входные данные - список
    if not isinstance(list_data, list):
        logger.error('Ошибка: входные данные не являются списком')
        return

    # проверка, что все элементы в списке - строки
    if not all(isinstance(item, str) for item in list_data):
        logger.error('Ошибка: элементы списка не являются строками')
        return
        
    # код выводит список в консоль
    for item in list_data:
        print(item)
