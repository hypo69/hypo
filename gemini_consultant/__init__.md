def show_help():
    print("\nДоступные команды:")
    print("1. Запустить скрипт 1 — Запускает скрипт 1.")
    print("2. Запустить скрипт 2 — Запускает скрипт 2.")
    print("3. --help — Показать это меню.")
    print("4. exit — Выход из программы.\n")


def interactive_menu():
    print("Добро пожаловать! Выберите одну из команд:\n")
    while True:
        print("1. Запустить скрипт 1")
        print("2. Запустить скрипт 2")
        print("3. --help — Показать список команд.")
        print("4. exit — Выход из программы.")

        choice = input("Введите номер команды: ").strip()

        if choice == "1":
            script1()
        elif choice == "2":
            script2()
        elif choice == "3" or choice.lower() == "--help":
            show_help()
        elif choice.lower() == "exit":
            print("Выход из программы.")
            sys.exit()  # Правильный способ завершить программу
        else:
            print("Некорректный ввод. Пожалуйста, выберите одну из предложенных команд.")

    parser = argparse.ArgumentParser(description="Интерактивное меню для запуска скриптов и команд.")
    parser.add_argument("--help", action="store_true", help="Показать доступные опции и справочную информацию")
    args = parser.parse_args()

    if args.help:
        show_help()
    else:
        interactive_menu()


if __name__ == "__main__":
    main()

**Изменения и улучшения:**

* **Обработка ошибок:**  Добавлены `try...except` блоки в `script1()` и `script2()`.  Это важно, чтобы ваша программа не падала при ошибках внутри скриптов.  Теперь сообщения об ошибках будут выводиться пользователю.
* **Правильный выход:**  Вместо `break` в `interactive_menu()` теперь используется `sys.exit()`. Это  *необходимо*, чтобы корректно завершить программу после ввода "exit".  `break` просто завершал цикл `while`, но не выходил из программы.
* **Документация:** Небольшое улучшение структуры кода для лучшей читабельности.
