"""
Модуль для решения задачи 271 Project Euler: Суммы остатков.
===========================================================

Этот модуль вычисляет сумму N(p) для всех простых чисел p в диапазоне от 5 до заданного предела,
где N(p) - сумма решений уравнения n^3 ≡ 1 (mod p) для 0 < n < p.

Функции:
    - is_prime(n): Проверяет, является ли число n простым.
    - sum_of_residues(limit): Вычисляет сумму N(p) для всех простых p в заданном диапазоне.
"""

from src.logger.logger import logger
from math import sqrt

def is_prime(n: int) -> bool:
    """
    Проверяет, является ли число простым.

    :param n: Число для проверки.
    :type n: int
    :return: True, если число простое, иначе False.
    :rtype: bool
    """
    if n <= 1:
        return False
    if n <= 3:
        return True
    if n % 2 == 0 or n % 3 == 0:
        return False
    i = 5
    while i * i <= n:
        if n % i == 0 or n % (i + 2) == 0:
            return False
        i += 6
    return True

def sum_of_residues(limit: int) -> int:
    """
    Вычисляет сумму N(p) для всех простых p в заданном диапазоне.

    :param limit: Верхняя граница диапазона для поиска простых чисел (не включая).
    :type limit: int
    :return: Сумма всех N(p) для простых чисел p.
    :rtype: int
    """
    total_sum_N = 0
    try:
        # Цикл перебирает все числа в заданном диапазоне
        for p in range(5, limit):
            # Проверка числа на простоту
            if is_prime(p):
                sum_n = 0
                # Если p mod 3 = 2, то n=1 является единственным решением
                if p % 3 == 2:
                    sum_n = 1
                else:
                    # Если p mod 3 != 2, перебираем все значения n от 1 до p-1
                    for n in range(1, p):
                        # Проверка n^3 mod p == 1
                        if pow(n, 3, p) == 1:
                            sum_n += n
                # Добавление sum_n к total_sum_N
                total_sum_N += sum_n
    except Exception as e:
        logger.error(f'Произошла ошибка при вычислении суммы остатков: {e}')
        return 0
    return total_sum_N

limit = 10**7
result = sum_of_residues(limit)
print(result)
