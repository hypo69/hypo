def get_project_root(marker_files=('pyproject.toml', 'requirements.txt', '.git')) -> Path:
    """!
    Находит корневой каталог проекта, начиная с текущего файла и поднимаясь вверх по иерархии каталогов,
    пока не найдёт один из файлов или директорий из списка `marker_files`.

    Args:
        marker_files (tuple): Имена файлов или директорий, которые указывают на корневой каталог проекта.

    Returns:
        Path: Путь к корневому каталогу, если он найден, иначе - каталог, где расположен текущий скрипт.
    """
    __root__: Path 
    current_path = Path(__file__).resolve().parent
    __root__ = current_path
    for parent in [current_path] + list(current_path.parents):
        if any((parent / marker).exists() for marker in marker_files):
            __root__ = parent
            break
    if str(__root__) not in sys.path:
        sys.path.insert(0, str(__root__))
        
    return __root__

__root__: Path = get_project_root()
