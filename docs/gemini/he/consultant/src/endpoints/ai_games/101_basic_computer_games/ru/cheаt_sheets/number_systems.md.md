    total_apples2 = apples2 + pears2 * 3 + melons2 * 15 // 3 + baskets2 * 30

    if total_apples1 < total_apples2: # –ø—Ä–æ–≤–µ—Ä—è–µ–º –≤–æ–∑–º–æ–∂–Ω–æ—Å—Ç—å –≤—ã—á–∏—Ç–∞–Ω–∏—è
        return "–ù–µ–≤–æ–∑–º–æ–∂–Ω–æ –≤—ã—á–µ—Å—Ç—å"
    else:
        total_apples = total_apples1 - total_apples2

    # –í–æ–∑–≤—Ä–∞—â–∞–µ–º –Ω–æ—Ä–º–∞–ª–∏–∑–æ–≤–∞–Ω–Ω–æ–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–∏–µ —Å—É–º–º—ã —è–±–ª–æ–∫
    result_fruits = ""
    baskets = total_apples // 30
    result_fruits += "üß∫" * baskets
    total_apples %= 30
    melons = (total_apples*3) // 15
    result_fruits += "üçâ" * melons
    total_apples %= 15
    pears = total_apples // 3
    result_fruits += "üçê" * pears
    total_apples %= 3
    result_fruits += "üçé" * total_apples

    return normalize_fruits(result_fruits)

        raise ValueError("–ù–µ–¥–æ–ø—É—Å—Ç–∏–º—ã–π –≤–≤–æ–¥: –¥–µ—Å—è—Ç–∏—á–Ω–æ–µ —á–∏—Å–ª–æ –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å —Ü–µ–ª—ã–º –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–º —á–∏—Å–ª–æ–º") # –≤—ã–∫–∏–¥—ã–≤–∞–µ–º –∏—Å–∫–ª—é—á–µ–Ω–∏–µ
    if decimal == 0: # –µ—Å–ª–∏ –¥–µ—Å—è—Ç–∏—á–Ω–æ–µ —á–∏—Å–ª–æ 0
        return "0" # –≤–æ–∑–≤—Ä–∞—â–∞–µ–º 0
    septenary = ""  # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ–º —Å—Ç—Ä–æ–∫—É –¥–ª—è —Å–µ–º–µ—Ä–∏—á–Ω–æ–≥–æ —á–∏—Å–ª–∞
    while decimal > 0: # –ø–æ–∫–∞ —á–∏—Å–ª–æ > 0
        septenary = str(decimal % 7) + septenary # –¥–æ–±–∞–≤–ª—è–µ–º –æ—Å—Ç–∞—Ç–æ–∫ –æ—Ç –¥–µ–ª–µ–Ω–∏—è –Ω–∞ 7
        decimal = decimal // 7 # —Ü–µ–ª–æ—á–∏—Å–ª–µ–Ω–Ω–æ–µ –¥–µ–ª–µ–Ω–∏–µ
    return septenary # –≤–æ–∑–≤—Ä–∞—â–∞–µ–º —Å–µ–º–µ—Ä–∏—á–Ω—É—é —Å—Ç—Ä–æ–∫—É


def hex_to_dec(hexadecimal: str) -> int:
    """
    –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ—Ç —à–µ—Å—Ç–Ω–∞–¥—Ü–∞—Ç–µ—Ä–∏—á–Ω–æ–µ —á–∏—Å–ª–æ (—Å—Ç—Ä–æ–∫—É) –≤ –¥–µ—Å—è—Ç–∏—á–Ω–æ–µ.

    :param hexadecimal: –®–µ—Å—Ç–Ω–∞–¥—Ü–∞—Ç–µ—Ä–∏—á–Ω–æ–µ —á–∏—Å–ª–æ –≤ –≤–∏–¥–µ —Å—Ç—Ä–æ–∫–∏.
    :type hexadecimal: str
    :raises ValueError: –ï—Å–ª–∏ –≤—Ö–æ–¥–Ω–∞—è —Å—Ç—Ä–æ–∫–∞ —Å–æ–¥–µ—Ä–∂–∏—Ç –Ω–µ–¥–æ–ø—É—Å—Ç–∏–º—ã–µ —à–µ—Å—Ç–Ω–∞–¥—Ü–∞—Ç–µ—Ä–∏—á–Ω—ã–µ —Å–∏–º–≤–æ–ª—ã.
    :return: –î–µ—Å—è—Ç–∏—á–Ω–æ–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–∏–µ —á–∏—Å–ª–∞ (—Ü–µ–ª–æ–µ —á–∏—Å–ª–æ).
    :rtype: int

    –ü—Ä–∏–º–µ—Ä:
    
    >>> hex_to_dec("2AF")
    687
    """
    if not all(digit.upper() in '0123456789ABCDEF' for digit in hexadecimal): # –ø—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ—Å—Ç—å –≤—Ö–æ–¥–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö
        logger.error('–ù–µ–¥–æ–ø—É—Å—Ç–∏–º—ã–µ —Å–∏–º–≤–æ–ª—ã –≤–æ –≤—Ö–æ–¥–Ω–æ–π —à–µ—Å—Ç–Ω–∞–¥—Ü–∞—Ç–µ—Ä–∏—á–Ω–æ–π —Å—Ç—Ä–æ–∫–µ') # –ª–æ–≥–∏—Ä—É–µ–º –æ—à–∏–±–∫—É
        raise ValueError("–ù–µ–¥–æ–ø—É—Å—Ç–∏–º—ã–µ —Å–∏–º–≤–æ–ª—ã –≤–æ –≤—Ö–æ–¥–Ω–æ–π —à–µ—Å—Ç–Ω–∞–¥—Ü–∞—Ç–µ—Ä–∏—á–Ω–æ–π —Å—Ç—Ä–æ–∫–µ") # –≤—ã–∫–∏–¥—ã–≤–∞–µ–º –∏—Å–∫–ª—é—á–µ–Ω–∏–µ

    decimal = 0 # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ–º –¥–µ—Å—è—Ç–∏—á–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ
    power = 0 # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ–º —Å—Ç–µ–ø–µ–Ω—å 16 (–ø–æ–∫–∞–∑–∞—Ç–µ–ª—å —Ä–∞–∑—Ä—è–¥–∞)
    for digit in reversed(hexadecimal): # –ò—Ç–µ—Ä–∏—Ä—É–µ–º—Å—è –ø–æ —Ü–∏—Ñ—Ä–∞–º —à–µ—Å—Ç–Ω–∞–¥—Ü–∞—Ç–µ—Ä–∏—á–Ω–æ–≥–æ —á–∏—Å–ª–∞ –≤ –æ–±—Ä–∞—Ç–Ω–æ–º –ø–æ—Ä—è–¥–∫–µ
        if digit.isdigit(): # –µ—Å–ª–∏ —Ü–∏—Ñ—Ä–∞
            decimal += int(digit) * (16 ** power) # –¥–æ–±–∞–≤–ª—è–µ–º —Ü–∏—Ñ—Ä—É * 16 –≤ —Å—Ç–µ–ø–µ–Ω–∏ —Ä–∞–∑—Ä—è–¥–∞
        elif digit.upper() == 'A': # –µ—Å–ª–∏ A
            decimal += 10 * (16 ** power) # –¥–æ–±–∞–≤–ª—è–µ–º 10 * 16 –≤ —Å—Ç–µ–ø–µ–Ω–∏ —Ä–∞–∑—Ä—è–¥–∞
        elif digit.upper() == 'B': # –µ—Å–ª–∏ B
            decimal += 11 * (16 ** power) # –¥–æ–±–∞–≤–ª—è–µ–º 11 * 16 –≤ —Å—Ç–µ–ø–µ–Ω–∏ —Ä–∞–∑—Ä—è–¥–∞
        elif digit.upper() == 'C': # –µ—Å–ª–∏ C
            decimal += 12 * (16 ** power) # –¥–æ–±–∞–≤–ª—è–µ–º 12 * 16 –≤ —Å—Ç–µ–ø–µ–Ω–∏ —Ä–∞–∑—Ä—è–¥–∞
        elif digit.upper() == 'D': # –µ—Å–ª–∏ D
            decimal += 13 * (16 ** power) # –¥–æ–±–∞–≤–ª—è–µ–º 13 * 16 –≤ —Å—Ç–µ–ø–µ–Ω–∏ —Ä–∞–∑—Ä—è–¥–∞
        elif digit.upper() == 'E': # –µ—Å–ª–∏ E
            decimal += 14 * (16 ** power) # –¥–æ–±–∞–≤–ª—è–µ–º 14 * 16 –≤ —Å—Ç–µ–ø–µ–Ω–∏ —Ä–∞–∑—Ä—è–¥–∞
        elif digit.upper() == 'F': # –µ—Å–ª–∏ F
            decimal += 15 * (16 ** power) # –¥–æ–±–∞–≤–ª—è–µ–º 15 * 16 –≤ —Å—Ç–µ–ø–µ–Ω–∏ —Ä–∞–∑—Ä—è–¥–∞
        power += 1 # —É–≤–µ–ª–∏—á–∏–≤–∞–µ–º —Å—Ç–µ–ø–µ–Ω—å
    return decimal # –≤–æ–∑–≤—Ä–∞—â–∞–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç


def dec_to_hex(decimal: int) -> str:
    """
    –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ—Ç –¥–µ—Å—è—Ç–∏—á–Ω–æ–µ —á–∏—Å–ª–æ (—Ü–µ–ª–æ–µ) –≤ —à–µ—Å—Ç–Ω–∞–¥—Ü–∞—Ç–µ—Ä–∏—á–Ω–æ–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–∏–µ (—Å—Ç—Ä–æ–∫—É).

    :param decimal: –î–µ—Å—è—Ç–∏—á–Ω–æ–µ —á–∏—Å–ª–æ.
    :type decimal: int
    :raises ValueError: –ï—Å–ª–∏ –≤—Ö–æ–¥–Ω–æ–µ —á–∏—Å–ª–æ –Ω–µ —è–≤–ª—è–µ—Ç—Å—è —Ü–µ–ª—ã–º –∏–ª–∏ –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–º.

    :rtype: str

    –ü—Ä–∏–º–µ—Ä:

    >>> dec_to_hex(687)
    '2AF'
    """
    if not isinstance(decimal, int) or decimal < 0: # –ø—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ—Å—Ç—å –≤—Ö–æ–¥–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö
        logger.error('–ù–µ–¥–æ–ø—É—Å—Ç–∏–º—ã–π –≤–≤–æ–¥: –¥–µ—Å—è—Ç–∏—á–Ω–æ–µ —á–∏—Å–ª–æ –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å —Ü–µ–ª—ã–º –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–º —á–∏—Å–ª–æ–º') # –ª–æ–≥–∏—Ä—É–µ–º –æ—à–∏–±–∫—É
        raise ValueError("–ù–µ–¥–æ–ø—É—Å—Ç–∏–º—ã–π –≤–≤–æ–¥: –¥–µ—Å—è—Ç–∏—á–Ω–æ–µ —á–∏—Å–ª–æ –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å —Ü–µ–ª—ã–º –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–º —á–∏—Å–ª–æ–º") # –≤—ã–∫–∏–¥—ã–≤–∞–µ–º –∏—Å–∫–ª—é—á–µ–Ω–∏–µ
    if decimal == 0:  # –ï—Å–ª–∏ –¥–µ—Å—è—Ç–∏—á–Ω–æ–µ —á–∏—Å–ª–æ —Ä–∞–≤–Ω–æ 0
        return "0" # –í–æ–∑–≤—Ä–∞—â–∞–µ–º —Å—Ç—Ä–æ–∫—É "0"
    hex_digits = "0123456789ABCDEF"  # –°—Ç—Ä–æ–∫–∞ –¥–ª—è —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏—è –æ—Å—Ç–∞—Ç–∫–æ–≤ –∏ —à–µ—Å—Ç–Ω–∞–¥—Ü–∞—Ç–µ—Ä–∏—á–Ω—ã—Ö —Ü–∏—Ñ—Ä
    hexadecimal = "" # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ–º —Å—Ç—Ä–æ–∫—É –¥–ª—è —à–µ—Å—Ç–Ω–∞–¥—Ü–∞—Ç–µ—Ä–∏—á–Ω–æ–≥–æ —á–∏—Å–ª–∞
    while decimal > 0: # –ü–æ–∫–∞ –¥–µ—Å—è—Ç–∏—á–Ω–æ–µ —á–∏—Å–ª–æ –±–æ–ª—å—à–µ 0
        remainder = decimal % 16 # –ü–æ–ª—É—á–∞–µ–º –æ—Å—Ç–∞—Ç–æ–∫ –æ—Ç –¥–µ–ª–µ–Ω–∏—è –Ω–∞ 16
        hexadecimal = hex_digits[remainder] + hexadecimal # –î–æ–±–∞–≤–ª—è–µ–º —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â—É—é —Ü–∏—Ñ—Ä—É –≤ –Ω–∞—á–∞–ª–æ —à–µ—Å—Ç–Ω–∞–¥—Ü–∞—Ç–µ—Ä–∏—á–Ω–æ–π —Å—Ç—Ä–æ–∫–∏
        decimal = decimal // 16 # –¶–µ–ª–æ—á–∏—Å–ª–µ–Ω–Ω–æ –¥–µ–ª–∏–º –¥–µ—Å—è—Ç–∏—á–Ω–æ–µ —á–∏—Å–ª–æ –Ω–∞ 16
    return hexadecimal # –í–æ–∑–≤—Ä–∞—â–∞–µ–º —à–µ—Å—Ç–Ω–∞–¥—Ü–∞—Ç–µ—Ä–∏—á–Ω—É—é —Å—Ç—Ä–æ–∫—É


def roman_to_int(roman_str: str) -> int:
    """
    –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ—Ç —Ä–∏–º—Å–∫–æ–µ —á–∏—Å–ª–æ (—Å—Ç—Ä–æ–∫—É) –≤ –¥–µ—Å—è—Ç–∏—á–Ω–æ–µ.

    :param roman_str: –†–∏–º—Å–∫–æ–µ —á–∏—Å–ª–æ –≤ –≤–∏–¥–µ —Å—Ç—Ä–æ–∫–∏.
    :type roman_str: str
    :raises ValueError: –ï—Å–ª–∏ –≤—Ö–æ–¥–Ω–∞—è —Å—Ç—Ä–æ–∫–∞ —Å–æ–¥–µ—Ä–∂–∏—Ç –Ω–µ–¥–æ–ø—É—Å—Ç–∏–º—ã–µ —Ä–∏–º—Å–∫–∏–µ —Å–∏–º–≤–æ–ª—ã.
    :return: –î–µ—Å—è—Ç–∏—á–Ω–æ–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–∏–µ —á–∏—Å–ª–∞ (—Ü–µ–ª–æ–µ —á–∏—Å–ª–æ).
    :rtype: int

    –ü—Ä–∏–º–µ—Ä:

    >>> roman_to_int("XIV")
    14
    """
    roman_dict = {
        'I': 1,
        'V': 5,
        'X': 10,
        'L': 50,
        'C': 100,
        'D': 500,
        'M': 1000
    }
    if not all(char in roman_dict for char in roman_str): # –ø—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ—Å—Ç—å –≤—Ö–æ–¥–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö
        logger.error('–ù–µ–¥–æ–ø—É—Å—Ç–∏–º—ã–µ —Å–∏–º–≤–æ–ª—ã –≤–æ –≤—Ö–æ–¥–Ω–æ–π —Ä–∏–º—Å–∫–æ–π —Å—Ç—Ä–æ–∫–µ') # –ª–æ–≥–∏—Ä—É–µ–º –æ—à–∏–±–∫—É
        raise ValueError("–ù–µ–¥–æ–ø—É—Å—Ç–∏–º—ã–µ —Å–∏–º–≤–æ–ª—ã –≤–æ –≤—Ö–æ–¥–Ω–æ–π —Ä–∏–º—Å–∫–æ–π —Å—Ç—Ä–æ–∫–µ") # –≤—ã–∫–∏–¥—ã–≤–∞–µ–º –∏—Å–∫–ª—é—á–µ–Ω–∏–µ

    number = 0
    roman_str = roman_str.replace("IV","IIII") # –∑–∞–º–µ–Ω—è–µ–º IV –Ω–∞ IIII
    roman_str = roman_str.replace("IX","VIIII") # –∑–∞–º–µ–Ω—è–µ–º IX –Ω–∞ VIIII
    roman_str = roman_str.replace("XL","XXXX") # –∑–∞–º–µ–Ω—è–µ–º XL –Ω–∞ XXXX
    roman_str = roman_str.replace("XC","LXXXX") # –∑–∞–º–µ–Ω—è–µ–º XC –Ω–∞ LXXXX
    roman_str = roman_str.replace("CD","CCCC") # –∑–∞–º–µ–Ω—è–µ–º CD –Ω–∞ CCCC
    roman_str = roman_str.replace("CM","DCCCC") # –∑–∞–º–µ–Ω—è–µ–º CM –Ω–∞ DCCCC
    for char in roman_str: # –ø—Ä–æ—Ö–æ–¥–∏–º –ø–æ —Å—Ç—Ä–æ–∫–µ
        number += roman_dict[char] # –¥–æ–±–∞–≤–ª—è–µ–º –∑–Ω–∞—á–µ–Ω–∏–µ —Å–∏–º–≤–æ–ª–∞
    return number # –≤–æ–∑–≤—Ä–∞—â–∞–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç

    """

def morse_to_sound(morse_code: str):
    """
    –í–æ—Å–ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç –∫–æ–¥ –ú–æ—Ä–∑–µ –≤ –≤–∏–¥–µ –∑–≤—É–∫–æ–≤—ã—Ö —Å–∏–≥–Ω–∞–ª–æ–≤.

    for symbol in morse_code: # –∏—Ç–µ—Ä–∏—Ä—É–µ–º—Å—è –ø–æ —Å–∏–º–≤–æ–ª–∞–º

def calculate_day_of_week(start_day: int, days_passed: float) -> int:
    """
    –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ—Ç –¥–µ–Ω—å –Ω–µ–¥–µ–ª–∏ –ø–æ—Å–ª–µ –∑–∞–¥–∞–Ω–Ω–æ–≥–æ –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ –¥–Ω–µ–π.

    :type start_day: int
    :param days_passed: –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –ø—Ä–æ—à–µ–¥—à–∏—Ö –¥–Ω–µ–π.

    :return: –î–µ–Ω—å –Ω–µ–¥–µ–ª–∏ –ø–æ—Å–ª–µ –∑–∞–¥–∞–Ω–Ω–æ–≥–æ –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ –¥–Ω–µ–π (0 - –ø–æ–Ω–µ–¥–µ–ª—å–Ω–∏–∫, 6 - –≤–æ—Å–∫—Ä–µ—Å–µ–Ω—å–µ).
    :rtype: int

    –ü—Ä–∏–º–µ—Ä:

    >>> calculate_day_of_week(0, 8)
    1
    """
    if not isinstance(start_day, int) or not (0 <= start_day <= 6): # –ø—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ—Å—Ç—å –≤—Ö–æ–¥–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö
        logger.error("–ù–∞—á–∞–ª—å–Ω—ã–π –¥–µ–Ω—å –Ω–µ–¥–µ–ª–∏ –¥–æ–ª–∂–µ–Ω –±—ã—Ç—å —Ü–µ–ª—ã–º —á–∏—Å–ª–æ–º –æ—Ç 0 –¥–æ 6 (–ø–Ω-–≤—Å)") # –ª–æ–≥–∏—Ä—É–µ–º –æ—à–∏–±–∫—É
        raise ValueError("–ù–∞—á–∞–ª—å–Ω—ã–π –¥–µ–Ω—å –Ω–µ–¥–µ–ª–∏ –¥–æ–ª–∂–µ–Ω –±—ã—Ç—å —Ü–µ–ª—ã–º —á–∏—Å–ª–æ–º –æ—Ç 0 –¥–æ 6 (–ø–Ω-–≤—Å)") # –≤—ã–∫–∏–¥—ã–≤–∞–µ–º –∏—Å–∫–ª—é—á–µ–Ω–∏–µ
    if not isinstance(days_passed, (int, float)): # –ø—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ—Å—Ç—å –≤—Ö–æ–¥–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö
        logger.error("–ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –ø—Ä–æ—à–µ–¥—à–∏—Ö –¥–Ω–µ–π –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å —á–∏—Å–ª–æ–º") # –ª–æ–≥–∏—Ä—É–µ–º –æ—à–∏–±–∫—É
        raise ValueError("–ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –ø—Ä–æ—à–µ–¥—à–∏—Ö –¥–Ω–µ–π –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å —á–∏—Å–ª–æ–º") # –≤—ã–∫–∏–¥—ã–≤–∞–µ–º –∏—Å–∫–ª—é—á–µ–Ω–∏–µ

    days_passed = int(days_passed) # –ø—Ä–µ–æ–±—Ä–∞–∑—É–µ–º –≤ —Ü–µ–ª–æ–µ
    new_day = (start_day + days_passed) % 7 # –≤—ã—á–∏—Å–ª—è–µ–º –¥–µ–Ω—å –Ω–µ–¥–µ–ª–∏
    return new_day # –≤–æ–∑–≤—Ä–∞—â–∞–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç
