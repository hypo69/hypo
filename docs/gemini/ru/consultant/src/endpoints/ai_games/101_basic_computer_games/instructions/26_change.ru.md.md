```python
"""
Модуль для реализации игры CHANGE (Кассовая программа).
=========================================================================================

Этот модуль содержит функции для имитации работы кассира в магазине,
вычисления сдачи и разбиения ее на банкноты и монеты.
"""

from src.logger.logger import logger # импорт logger для логирования ошибок

def calculate_change(price: float, payment: float) -> dict:
    """
    Вычисляет сдачу и разбивает ее на банкноты и монеты.

    :param price: Стоимость товара.
    :param payment: Сумма оплаты.
    :return: Словарь с количеством банкнот и монет каждого номинала.
    :raises ValueError: Если сумма оплаты меньше стоимости товара.

    Пример:
        >>> calculate_change(4.59, 10.00)
        {'five_dollar': 1, 'quarter': 1, 'nickel': 1, 'penny': 1}
    """
    if payment < price:
        logger.error(f'Сумма оплаты {payment} меньше стоимости товара {price}')
        raise ValueError('Сумма оплаты не может быть меньше стоимости товара')

    change = round(payment - price, 2) # расчет сдачи
    change_dict = {} # словарь для хранения результатов

    denominations = {  # словарь с номиналами банкнот и монет
        'five_dollar': 5.00,
        'dollar': 1.00,
        'quarter': 0.25,
        'nickel': 0.05,
        'penny': 0.01
    }

    for name, value in denominations.items(): # перебор номиналов
        while change >= value: # пока сдача больше или равна номиналу
            change = round(change - value, 2) # уменьшение сдачи
            change_dict[name] = change_dict.get(name, 0) + 1  # добавление номинала в словарь

    return change_dict  # возврат словаря с номиналами


def print_change(change_dict: dict, total_change: float) -> None:
    """
    Выводит на экран сдачу, разложенную по банкнотам и монетам.

    :param change_dict: Словарь с количеством банкнот и монет каждого номинала.
    :param total_change: Общая сумма сдачи.
    """
    print(f"Сдача: ${total_change}") # вывод общей суммы сдачи

    for name, count in change_dict.items(): # перебор номиналов
        if count > 0:
            print(f"{count} {name.replace('_', ' ')} {'купюра' if 'dollar' in name else 'монета' if 'penny' in name else 'монеты' if count >1 else 'монета'}") # вывод номинала и его количества



def play_change_game() -> None:
    """
    Запускает игру CHANGE.
    """
    print("Добро пожаловать в игру CHANGE!") # приветствие игрока
    while True: # начало цикла игры
        try: # обработка ошибок
            price = float(input("Введите стоимость товара: ")) # ввод стоимости товара
            payment = float(input("Введите сумму оплаты: ")) # ввод суммы оплаты

            change_dict = calculate_change(price, payment) # вычисление сдачи
            total_change = round(payment - price, 2) # расчет общей сдачи

            print_change(change_dict, total_change)  # вывод сдачи на экран

        except ValueError as e:
            logger.error(f'Ошибка ввода данных или расчета сдачи: {e}') # логирование ошибок
            print('Произошла ошибка. Пожалуйста, проверьте введенные данные.') # сообщение об ошибке
            continue # переход к следующей итерации цикла
        except Exception as ex: # обработка любых других ошибок
             logger.error(f'Непредвиденная ошибка: {ex}') # логирование ошибок
             print(f'Непредвиденная ошибка: {ex}')
             continue # переход к следующей итерации цикла

        play_again = input("Хотите сыграть снова? (да/нет) ") # запрос на повтор игры
        if play_again.lower() != 'да':  # если игрок не хочет играть снова, выход из цикла
            print("Спасибо за игру!") # прощание с игроком
            break


if __name__ == "__main__":
    play_change_game() # запуск игры
```
