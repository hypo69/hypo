"""
Модуль для решения задачи 844 Project Euler.
==================================================

Этот модуль вычисляет сумму квадратов первых n простых чисел по модулю.
Использует алгоритм решета Эратосфена для генерации простых чисел.

Пример использования
--------------------

.. code-block:: python

   result = sum_of_squares_of_primes(100000000, 1000000007)
   print(result)

"""
from src.logger.logger import logger

def is_prime(number: int) -> bool:
    """
    Проверяет, является ли число простым.

    :param number: Число для проверки.
    :return: True, если число простое, иначе False.
    """
    if number < 2:
        return False
    for i in range(2, int(number**0.5) + 1):
        if number % i == 0:
            return False
    return True


def sum_of_squares_of_primes(n: int, modulus: int) -> int:
    """
    Вычисляет сумму квадратов первых n простых чисел по модулю.

    :param n: Количество простых чисел для суммирования.
    :param modulus: Модуль, по которому нужно вычислять сумму.
    :return: Сумма квадратов первых n простых чисел по модулю.
    """
    if not isinstance(n, int) or n <= 0:
        logger.error(f'Некорректный ввод: n должен быть положительным целым числом, получено: {n}')
        return 0
    if not isinstance(modulus, int) or modulus <= 0:
        logger.error(f'Некорректный ввод: modulus должен быть положительным целым числом, получено: {modulus}')
        return 0
    
    primes = []
    number = 2
    while len(primes) < n:
        if is_prime(number):
            primes.append(number)
        number += 1
    
    result = 0
    for prime in primes:
        result = (result + prime**2) % modulus
    
    return result

if __name__ == "__main__":
    """
    Пример использования функции.
    """
    n = 10**8
    modulus = 10**9 + 7
    result = sum_of_squares_of_primes(n, modulus)
    print(f'S({n}) mod {modulus} = {result}')
