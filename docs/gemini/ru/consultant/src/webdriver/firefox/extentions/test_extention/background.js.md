```python
"""
Модуль background.js для Firefox расширения.
=========================================================================================

Этот модуль прослушивает событие нажатия на иконку расширения и внедряет content script в активную вкладку.
"""
from src.logger.logger import logger

# Слушаем событие нажатия на иконку расширения
browser.browserAction.onClicked.addListener(async (tab) => {
    """
    Функция-обработчик события нажатия на иконку расширения.

    Args:
        tab (object): Объект вкладки, на которой произошло нажатие.
    """
    try:
        # Код выполняет запуск скрипта contentScript.js на текущей вкладке
        await browser.scripting.executeScript({
            'target': { 'tabId': tab.id },
            'files': ['contentScript.js'],
        });
    except Exception as ex:
        # Логирование ошибки, если скрипт не удалось выполнить
        logger.error(f"Ошибка при выполнении скрипта contentScript.js: {ex}")
});
```
