```python
# -*- coding: utf-8 -*-

"""
Модуль для установки путей и переменных окружения.
==================================================

Этот модуль определяет корневую директорию проекта, добавляет ее в `sys.path`,
а также добавляет рабочую директорию в `sys.path`.
"""

import os
import sys
from pathlib import Path
# импортируем logger
from src.logger.logger import logger

# Получаем корневую директорию проекта, обрезая путь до директории hypotez
dir_root: Path = Path(os.getcwd()[:os.getcwd().rfind('hypotez') + 7])
# Добавляем корневую директорию в sys.path для импорта модулей проекта
sys.path.append(str(dir_root))
#  Определяем директорию src
dir_src = Path(dir_root, 'src')
# Добавляем рабочую директорию в sys.path для импорта модулей проекта
sys.path.append(str(dir_src))
```
