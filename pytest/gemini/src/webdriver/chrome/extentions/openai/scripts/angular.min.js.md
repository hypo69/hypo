def test_De():
  """Test De, get attribute with specific prefix."""
  mock_element = MagicMock(hasAttribute = lambda x : True, getAttribute = lambda x : x)
  assert angular.De(mock_element, 'test') == 'ng-apptest'
  mock_element.hasAttribute = lambda x : False
  assert angular.De(mock_element, 'test') == None

def test_Ee_with_strict_di_enabled():
    """Test Ee when strict-di is enabled and the document and app attribute are present."""
    mock_element = MagicMock(
        hasAttribute = lambda x : True,
        getAttribute = lambda x : x
        )
    mock_bootstrap_func = MagicMock()
    angular.Ee(mock_element, mock_bootstrap_func)
    mock_bootstrap_func.assert_called_with(mock_element, ['ng-appapp'], {'strictDi': True})

def test_Ee_with_strict_di_disabled():
  """Test Ee when strict-di is disabled because of mismatch between document.location.href and script url"""
  mock_element = MagicMock(
    hasAttribute = lambda x : True,
    getAttribute = lambda x : x,
        querySelector = lambda x : x
    )
  window.location = MagicMock(origin="notmatching")
  mock_bootstrap_func = MagicMock()
  with pytest.raises(Exception) as excinfo:
     angular.Ee(mock_element, mock_bootstrap_func)
  assert "AngularJS: disabling automatic bootstrap" in str(excinfo)
  window.location = MagicMock(origin="http://localhost:8080")

def test_Ee_with_no_element_with_app_attribute():

    mock_bootstrap_func = MagicMock()
    angular.Ee(mock_element, mock_bootstrap_func)
