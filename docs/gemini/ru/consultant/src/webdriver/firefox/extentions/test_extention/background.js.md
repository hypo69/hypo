"""
Модуль background.js для управления расширением браузера.
========================================================

Этот модуль содержит обработчик события клика на иконку расширения
и выполняет скрипт contentScript.js на активной вкладке.

Пример использования
--------------------

При клике на иконку расширения, скрипт contentScript.js
будет выполнен на текущей вкладке.
"""
from src.logger.logger import logger

# Обработчик события клика на иконку расширения
browser.browserAction.onClicked.addListener(async (tab) => {
    """
    Добавляет слушателя на событие клика по иконке расширения.

    При клике на иконку, запускает скрипт `contentScript.js` на активной вкладке.

    :param tab: Объект вкладки, на которой произошло событие клика.
    """
    try:
        # Код исполняет запуск скрипта contentScript.js на текущей вкладке.
        await browser.scripting.executeScript({
            'target': {'tabId': tab.id},
            'files': ['contentScript.js'],
        })
    except Exception as ex:
        # Обработка ошибки при выполнении скрипта и логирование ее
        logger.error('Ошибка при выполнении скрипта contentScript.js', ex)
        ...
});
