    Включает в себя функции для нормализации названий продуктов и логических значений.
"""

from typing import Dict, Union
from src.utils.string import StringNormalizer as sn
from src.utils.string import StringFormatter as sf

def normalize_product_name(value: str) -> str:
    """Нормализует имя продукта.

    Args:
        value (str): Имя продукта.

    Returns:
        str: Нормализованное имя продукта.  Возвращает строку, очищенную от ненужных символов и приведённую к нижнему регистру.
    """
    return sn.normalize_string(value)

def normalize_bool(value: Union[str, bool]) -> int:
    """Преобразует логическое значение в 1/0.

    Args:
        value (Union[str, bool]): Значение, которое нужно преобразовать. Может быть строкой 'true'/'false' или булевым значением.

    Returns:
        int: 1 для True, 0 для False.  Возвращает целое число.
    """
    return 1 if sn.normalize_boolean(value) else 0
