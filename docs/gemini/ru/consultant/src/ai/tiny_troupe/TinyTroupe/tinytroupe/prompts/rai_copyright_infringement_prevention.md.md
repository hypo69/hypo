и информации с веб-сайтов, которые могут нарушать авторские права.
"""
from src.logger.logger import logger

def prevent_copyright_infringement(user_request):
    """
    Обрабатывает запрос пользователя на контент, защищенный авторским правом.

    :param user_request: Текст запроса пользователя.
    :return: Строка с отказом от предоставления контента.
    """
    # Проверка запроса на контент, защищенный авторским правом.
    # Находим ключевые слова, указывающие на запрос контента,
    # защищенного авторским правом
    keywords = ["книги", "песни", "рецепты", "новостные статьи", "WebMD"]
    if any(keyword in user_request.lower() for keyword in keywords):

        # Получаем краткое описание или сводку запрошенного материала
        # (можно улучшить, например, извлекая информацию из контекста запроса)
        try:
            summary = user_request
        except Exception as ex:
            logger.error("Ошибка получения сводки запрошенного материала", ex)
            summary = "Не удалось получить сводку."

        # Формируем отказ с указанием причин.

            "Извините, я не могу предоставить вам этот контент. "
            "Он защищен авторским правом, и я не имею права нарушать его. "
            f"Запрос: {summary}"
        )
        return response
    else:
        return None # Возвращаем None, если запрос не касается авторских прав
