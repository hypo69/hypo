 """
 Модуль для реализации текстовой игры "Футбол"
 =========================================================================================
 
 Данный модуль содержит логику игры "Футбол", в которой два игрока по очереди вводят числа,
 представляющие собой попытки ударов по воротам. Цель игры - забить гол противнику.
 
 Пример использования
 --------------------
 
 .. code-block:: python
 
     import footbl
 
     footbl.play_game()
 """
 from src.logger.logger import logger  # Импортируем logger для логирования ошибок
 
 
 def play_game():
     """
     Запускает игру "Футбол".
 
     Игра продолжается до тех пор, пока один из игроков не забьет гол.
     """
     # Инициализация текущего игрока (начинает игрок 1)
     currentPlayer = 1  # Устанавливаем начальное значение переменной `currentPlayer` в 1, что означает, что первый игрок начинает игру.
     
     # Основной игровой цикл
     while True:  # Бесконечный цикл, который продолжается до тех пор, пока один из игроков не забьет гол
         # Вывод сообщения о ходе текущего игрока
         print(f"PLAYER {currentPlayer} --- YOUR SHOT")  # Выводим сообщение, указывающее, чей сейчас ход.
         
         # Запрос ввода числа от игрока
         try:  # Блок try-except обрабатывает возможные ошибки ввода.
             shot = int(input("Введите число от 1 до 10: "))  # Запрашиваем ввод числа от текущего игрока и преобразуем введенную строку в целое число.
         except ValueError as e:  # Если пользователь введет не целое число, то будет выведено сообщение об ошибке.
             logger.error('Некорректный ввод числа', exc_info=True)  # Выводим сообщение об ошибке
             print("Пожалуйста, введите целое число.") # Сообщение пользователю о необходимости ввода целого числа
             continue  # Переходим к следующей итерации цикла
             
         # Проверка, был ли забит гол (если shot != 1)
         if shot != 1:  # Проверяем, не равно ли введенное число 1. Если число не равно 1, это считается голом.
             print(f"GOAL!!! PLAYER {currentPlayer} WINS!!!")  # Выводим сообщение о том, что игрок забил гол и выиграл игру.
             break  # Завершаем игровой цикл и игру.
         
         # Смена игрока: если был игрок 1, то теперь 2, и наоборот
         currentPlayer = 3 - currentPlayer  # Меняем текущего игрока на противоположного. Если `currentPlayer` был 1, то он станет 2, и наоборот.
 
 if __name__ == '__main__':
     play_game()
 """
 Объяснение кода:
 1.  **Инициализация текущего игрока**:
     -   `currentPlayer = 1`: Устанавливает начальное значение переменной `currentPlayer` в 1, что означает, что первый игрок начинает игру.
 
 2.  **Основной игровой цикл `while True:`**:
     -   Бесконечный цикл, который продолжается до тех пор, пока один из игроков не забьет гол (не будет выполнен оператор `break`).
 
 3.  **Вывод сообщения о текущем игроке**:
     -   `print(f"PLAYER {currentPlayer} --- YOUR SHOT")`: Выводит сообщение, указывающее, чей сейчас ход.
 
 4.  **Запрос ввода числа от игрока**:
     -   `try...except ValueError`: Блок try-except обрабатывает возможные ошибки ввода. Если пользователь введет не целое число, то будет выведено сообщение об ошибке.
     -   `shot = int(input("Введите число от 1 до 10: "))`: Запрашивает ввод числа от текущего игрока и преобразует введенную строку в целое число. Это число представляет собой попытку удара.
 
 5.  **Проверка, был ли забит гол**:
     -   `if shot != 1:`: Проверяет, не равно ли введенное число 1. Если число не равно 1, это считается голом.
     -   `print(f"GOAL!!! PLAYER {currentPlayer} WINS!!!")`: Выводит сообщение о том, что игрок забил гол и выиграл игру.
     -   `break`: Завершает игровой цикл и игру.
 
 6.  **Смена игрока**:
     -   `currentPlayer = 3 - currentPlayer`: Меняет текущего игрока на противоположного. Если `currentPlayer` был 1, то он станет 2, и наоборот.
 """
