```python
## \file /src/suppliers/cdata/login.py
# -*- coding: utf-8 -*-

"""
Модуль для авторизации в системе C-Data с использованием веб-драйвера.

Модуль содержит функцию `login`, которая выполняет вход в систему C-Data
с использованием предоставленных учетных данных.

Пример использования:
----------------------
>>> from src.suppliers.cdata.login import login
>>> # Предположим, что self - это экземпляр класса, содержащий необходимые атрибуты
>>> # result = login(self)
>>> # if result:
>>> #     print('Успешная авторизация')
"""

from src.logger import logger  # Импорт модуля logger


def login(self) -> bool:
    """
    Выполняет авторизацию в системе C-Data.

    Args:
        self: Экземпляр класса, содержащий атрибуты для авторизации.

    Returns:
        bool: True в случае успешной авторизации, False в противном случае.
    
    Raises:
        Exception: В случае возникновения ошибки при авторизации.
    """
    try:
        self.get_url('https://reseller.c-data.co.il/Login')

        email = self.locators['login']['email']  # Получаем email
        password = self.locators['login']['password']  # Получаем пароль

        email_locator = (self.locators['login']['email_locator']['by'],
                           self.locators['login']['email_locator']['selector'])  # Локатор поля email

        password_locator = (self.locators['login']['password_locator']['by'],
                               self.locators['login']['password_locator']['selector'])  # Локатор поля пароля

        loginbutton_locator = (self.locators['login']['loginbutton_locator']['by'],
                                  self.locators['login']['loginbutton_locator']['selector'])  # Локатор кнопки логина

        logger.info(f''' email_locator {email_locator}
                    password_locator {password_locator}
                    loginbutton_locator {loginbutton_locator}''')  # Логируем локаторы

        self.find(email_locator).send_keys(email)  # Вводим email
        self.find(password_locator).send_keys(password)  # Вводим пароль
        self.find(loginbutton_locator).click()  # Кликаем на кнопку логина
        logger.info('C-data logged in')  # Логируем успешный вход
        return True
    except Exception as ex:
        logger.error('Error while logging in', ex, exc_info=True)  # Логируем ошибку
        return False
```
