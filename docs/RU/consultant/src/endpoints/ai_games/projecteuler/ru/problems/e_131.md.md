"""
Модуль для решения задачи 131 Project Euler - "Кубические простые числа".
=====================================================================

Этот модуль содержит функцию для подсчета количества простых чисел p < 1000000,
для которых существует такое целое n, что n³+n²+n+1 делится на p.

Функция использует математические свойства уравнения n²+n=pq-1 для поиска
таких простых чисел.

Пример использования
--------------------

.. code-block:: python

    result = count_primes_for_cubic_division(1000000)
    print(f"Количество простых чисел: {result}")
"""
from math import sqrt
from src.logger.logger import logger
from typing import List

def is_prime(n: int) -> bool:
    """
    Проверяет, является ли число простым.

    :param n: Число для проверки.
    :return: True, если число простое, False в противном случае.
    """
    if n <= 1:
        return False
    if n <= 3:
        return True
    if n % 2 == 0 or n % 3 == 0:
        return False
    i = 5
    while i * i <= n:
        if n % i == 0 or n % (i + 2) == 0:
            return False
        i += 6
    return True

def find_primes_with_cubic_property(limit: int) -> List[int]:
    """
    Находит простые числа p < limit, для которых существует такое целое n, что n³+n²+n+1 делится на p.

    :param limit: Верхняя граница для поиска простых чисел.
    :return: Список найденных простых чисел.
    """
    primes = []
    for p in range(2, limit):
        if is_prime(p):
            n_found = False
            for q in range(2, int(sqrt(limit))+1): # оптимизация - проверяем только простые q
                if is_prime(q) and (1 + 4 * (q*p -1)) > 0 : # добавляем условие для избежания отрицательного дискриминанта
                    discriminant = 1 + 4 * (q*p - 1)
                    sqrt_discriminant = sqrt(discriminant)
                    if sqrt_discriminant == int(sqrt_discriminant):
                        n = (-1 + int(sqrt_discriminant)) / 2
                        if n == int(n):
                            n_found = True
                            break
            if n_found:
                primes.append(p)
    return primes

def count_primes_for_cubic_division(limit: int) -> int:
    """
     Считает количество простых чисел p < limit, для которых существует такое целое n, что n³+n²+n+1 делится на p.

    :param limit: Верхняя граница для поиска простых чисел.
    :return: Количество найденных простых чисел.
    """
    try:
        primes = find_primes_with_cubic_property(limit)
        return len(primes)
    except Exception as ex:
         logger.error(f'Произошла ошибка при вычислении количества простых чисел: {ex}')
         return 0 # возвращаем 0 при ошибке

if __name__ == '__main__':
    limit = 1000000
    result = count_primes_for_cubic_division(limit)
    print(f"Количество простых чисел p < {limit}, удовлетворяющих условию задачи: {result}")
