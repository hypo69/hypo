```python
"""
Модуль для демонстрации различий между списками и словарями.
=============================================================

В этом модуле представлены примеры использования списков и словарей для представления данных о товарах.
Словарь используется для хранения характеристик одного товара, а список — для хранения набора товаров.

Примеры использования
--------------------
Примеры демонстрируют создание и манипуляции со списками и словарями для хранения информации о товарах.
"""
# Словарь (`dict`) – это идеальный способ представления *характеристик* одного товара, где есть пары "ключ-значение".
# Список (`list`) – это отличный способ представления *набора* однотипных товаров, где каждый товар может быть представлен словарем.

# 1. Словарь (`dict`) для представления характеристик товара
# Словарь для представления характеристик одного товара. В этом случае, ключами словаря будут названия характеристик, а значениями – их соответствующие значения.
product: dict = {
    "id": "12345",
    "name": "Смартфон SuperPhone X",
    "brand": "SuperTech",
    "price": 999.99,
    "color": "Space Gray",
    "screen_size": 6.7,
    "storage": "256 GB",
    "camera": "108 MP",
    "os": "Android 13",
    "available": True
} # Словарь product хранит характеристики товара

print("--- Характеристики товара ---") # Вывод заголовка характеристик товара
for key, value in product.items(): # Перебор элементов словаря product
    print(f"{key}: {value}") # Вывод ключа и значения характеристики товара

print("\n--- Доступ к конкретным характеристикам ---") # Вывод заголовка для доступа к характеристикам товара
print(f"Название: {product['name']}") # Вывод названия товара
print(f"Цена: ${product['price']}") # Вывод цены товара
print(f"Доступен: {'да' if product['available'] else 'нет'}") # Вывод доступности товара

# Разъяснение кода:
# * словарь `product`, где ключами являются названия характеристик товара (id, name, brand, price и т.д.), а значениями — их соответствующие значения.
# * `product.items()` для итерации по всем парам "ключ-значение" и вывел их на экран.

# 2. Список (`list`) для представления товаров в категории
# Список для представления списка товаров в определенной категории.
# В этом случае, каждый элемент списка будет представлять отдельный товар, который, в свою очередь, может быть представлен словарем.
products: list = [
    {
        "id": "12345",
        "name": "Смартфон SuperPhone X",
        "brand": "SuperTech",
        "price": 999.99,
        "available": True
    },
    {
        "id": "67890",
        "name": "Планшет TabPro S",
        "brand": "TechGenius",
        "price": 799.99,
        "available": False
    },
    {
        "id": "13579",
        "name": "Наушники SoundBeats Pro",
        "brand": "AudioMax",
        "price": 199.99,
        "available": True
    }
] # Список products хранит словари, каждый из которых представляет товар

print("\n--- Список товаров в категории ---") # Вывод заголовка списка товаров
for product in products: # Перебор списка товаров
   print(f"{product['name']} ({product['brand']}), Price: ${product['price']}, Available: {'Yes' if product['available'] else 'No'}") # Вывод информации о товаре

print("\n--- Доступ к первому товару ---") # Вывод заголовка доступа к первому товару
first_product: dict = products[0] # Получение первого товара из списка
print(f"ID: {first_product['id']}, Name: {first_product['name']}") # Вывод id и названия первого товара

print("\n--- Вывод только имен товаров ---") # Вывод заголовка для вывода только имен товаров
for product in products: # Перебор списка товаров
  print(f"Name: {product['name']}") # Вывод имени товара

# Разъяснение кода:
# * Список `products`, где каждый элемент - это словарь, представляющий отдельный товар.
# * Я проитерировал список и вывел на экран информацию о каждом товаре, используя ключи словаря для доступа к характеристикам товара.
# * Я показал, как получить доступ к первому товару в списке, используя индекс `0`.
# * Я проитерировал список еще раз, и показал как можно выводить только названия товаров.

```
