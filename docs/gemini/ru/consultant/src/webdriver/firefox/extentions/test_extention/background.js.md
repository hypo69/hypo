```python
"""
Модуль для обработки событий в фоновом режиме расширения.
========================================================

Этот модуль прослушивает событие нажатия на иконку расширения и выполняет скрипт ``contentScript.js`` на активной вкладке.

Пример использования
--------------------

Этот скрипт будет автоматически запущен браузером в фоновом режиме, и при нажатии на иконку расширения будет запущен скрипт ``contentScript.js``.
"""
from src.logger.logger import logger

# прослушивание события нажатия на иконку расширения
browser.action.onClicked.addListener(async (tab) => {
    """
    Обработчик события нажатия на иконку расширения.

    :param tab: Объект вкладки, на которой было произведено нажатие.
    """
    try:
        # выполнение скрипта contentScript.js на указанной вкладке
        await browser.scripting.executeScript({
            'target': {'tabId': tab.id},
            'files': ['contentScript.js'],
        });
    except Exception as ex:
        # логирование ошибок при выполнении скрипта
        logger.error('Ошибка при выполнении contentScript.js', exc_info=ex)
    }
});
```
