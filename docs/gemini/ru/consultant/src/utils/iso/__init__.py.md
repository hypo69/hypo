# -*- coding: utf-8 -*-

"""
Модуль для работы с ISO.
=========================================================================================

Этот модуль содержит основные определения и настройки для работы с кодировками ISO, 
необходимые для корректной обработки текстовых данных в различных форматах.

Описание
--------

Модуль обеспечивает поддержку различных кодировок ISO для унификации обработки текста.

Пример использования
--------------------

.. code-block:: python

    from src.utils.iso import check_iso_encoding
    
    is_valid = check_iso_encoding('some string')

"""
#   Импортируем logger для логирования
from src.logger.logger import logger
#   Функция проверки корректности iso кодировки. TODO добавить реализацию
def check_iso_encoding(value: str) -> bool:
    """
    Проверяет, является ли строка корректной в кодировке ISO.

    Args:
        value (str): Строка для проверки.

    Returns:
        bool: True, если строка корректна, False в противном случае.

    Example:
        >>> from src.utils.iso import check_iso_encoding
        >>> check_iso_encoding('тест')
        True
    """
    try:
        #   Код исполняет проверку кодировки строки
        value.encode('iso-8859-1')
        return True
    except UnicodeEncodeError:
        logger.error(f'Строка не соответствует кодировке iso-8859-1: {value}')
        return False
    except Exception as ex:
       logger.error(f'Непредвиденная ошибка при проверки кодировки: {ex}')
       return False
