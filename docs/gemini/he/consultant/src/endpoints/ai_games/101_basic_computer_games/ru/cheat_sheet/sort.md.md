"""

from typing import List, Tuple # –ò–º–ø–æ—Ä—Ç –Ω–µ–æ–±—Ö–æ–¥–∏–º—ã—Ö —Ç–∏–ø–æ–≤ –¥–ª—è –∞–Ω–Ω–æ—Ç–∞—Ü–∏–∏
from src.logger.logger import logger # –ò–º–ø–æ—Ä—Ç –ª–æ–≥–≥–µ—Ä–∞ –¥–ª—è –∑–∞–ø–∏—Å–∏ —Å–æ–±—ã—Ç–∏–π #

def compare_fruits(fruit1: Tuple[str, int], fruit2: Tuple[str, int]) -> int:
    """
    –°—Ä–∞–≤–Ω–∏–≤–∞–µ—Ç –¥–≤–∞ —Ñ—Ä—É–∫—Ç–∞ –ø–æ —Ä–∞–∑–º–µ—Ä—É.

    :param fruit1: –ö–æ—Ä—Ç–µ–∂ (—Ä–∞–∑–º–µ—Ä, –∏–¥–µ–Ω—Ç–∏—Ñ–∏–∫–∞—Ç–æ—Ä) –ø–µ—Ä–≤–æ–≥–æ —Ñ—Ä—É–∫—Ç–∞.
    :type fruit1: Tuple[str, int]
    :param fruit2: –ö–æ—Ä—Ç–µ–∂ (—Ä–∞–∑–º–µ—Ä, –∏–¥–µ–Ω—Ç–∏—Ñ–∏–∫–∞—Ç–æ—Ä) –≤—Ç–æ—Ä–æ–≥–æ —Ñ—Ä—É–∫—Ç–∞.
    :type fruit2: Tuple[str, int]
    :return: -1, –µ—Å–ª–∏ fruit1 –º–µ–Ω—å—à–µ fruit2, 1, –µ—Å–ª–∏ fruit1 –±–æ–ª—å—à–µ fruit2, 0, –µ—Å–ª–∏ —Ä–∞–≤–Ω—ã.
    :rtype: int

    .. code-block:: python

        compare_fruits(("üçé", 1), ("üçê", 2)) # –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç -1
        compare_fruits(("üçâ", 1), ("üçé", 2)) # –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç 1
        compare_fruits(("üçé", 1), ("üçé", 2)) # –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç 0
    """
    order = {"üçé": 0, "üçê": 1, "üçâ": 2, "üß∫": 3}  # –û–ø—Ä–µ–¥–µ–ª—è—é –ø–æ—Ä—è–¥–æ–∫ —Ñ—Ä—É–∫—Ç–æ–≤ –ø–æ —Ä–∞–∑–º–µ—Ä—É
    size1 = order.get(fruit1[0]) # –ü–æ–ª—É—á–∞—é —Ä–∞–∑–º–µ—Ä –ø–µ—Ä–≤–æ–≥–æ —Ñ—Ä—É–∫—Ç–∞
    size2 = order.get(fruit2[0]) # –ü–æ–ª—É—á–∞—é —Ä–∞–∑–º–µ—Ä –≤—Ç–æ—Ä–æ–≥–æ —Ñ—Ä—É–∫—Ç–∞
    if size1 < size2: # –ï—Å–ª–∏ —Ä–∞–∑–º–µ—Ä –ø–µ—Ä–≤–æ–≥–æ —Ñ—Ä—É–∫—Ç–∞ –º–µ–Ω—å—à–µ, –≤–æ–∑–≤—Ä–∞—â–∞—é -1
        return -1
    elif size1 > size2: # –ï—Å–ª–∏ —Ä–∞–∑–º–µ—Ä –ø–µ—Ä–≤–æ–≥–æ —Ñ—Ä—É–∫—Ç–∞ –±–æ–ª—å—à–µ, –≤–æ–∑–≤—Ä–∞—â–∞—é 1
        return 1
    else: # –ï—Å–ª–∏ —Ä–∞–∑–º–µ—Ä—ã —Ä–∞–≤–Ω—ã, –≤–æ–∑–≤—Ä–∞—â–∞—é 0
      return 0

def bubble_sort(fruits: List[Tuple[str, int]]) -> List[Tuple[str, int]]:
    """
    –°–æ—Ä—Ç–∏—Ä—É–µ—Ç —Å–ø–∏—Å–æ–∫ —Ñ—Ä—É–∫—Ç–æ–≤ –ø–æ —Ä–∞–∑–º–µ—Ä—É, –∏—Å–ø–æ–ª—å–∑—É—è –∞–ª–≥–æ—Ä–∏—Ç–º "–ø—É–∑—ã—Ä—å–∫–æ–≤–∞—è —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞".

    :param fruits: –°–ø–∏—Å–æ–∫ –∫–æ—Ä—Ç–µ–∂–µ–π (—Ä–∞–∑–º–µ—Ä, –∏–¥–µ–Ω—Ç–∏—Ñ–∏–∫–∞—Ç–æ—Ä).
    :type fruits: List[Tuple[str, int]]
    :return: –û—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã–π —Å–ø–∏—Å–æ–∫ –∫–æ—Ä—Ç–µ–∂–µ–π.
    :rtype: List[Tuple[str, int]]

    .. code-block:: python

        fruits = [("üçâ", 1), ("üçé", 2), ("üçê", 3)]
        bubble_sort(fruits) # –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç [("üçé", 2), ("üçê", 3), ("üçâ", 1)]
    """
    n = len(fruits)  # –ü–æ–ª—É—á–∞—é –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ñ—Ä—É–∫—Ç–æ–≤
    for i in range(n):  # –ü—Ä–æ—Ö–æ–∂—É –ø–æ —Å–ø–∏—Å–∫—É n —Ä–∞–∑
        for j in range(0, n - i - 1):  # –ü—Ä–æ—Ö–æ–∂—É –ø–æ –Ω–µ–æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω–æ–π —á–∞—Å—Ç–∏ —Å–ø–∏—Å–∫–∞
            if compare_fruits(fruits[j], fruits[j + 1]) == 1:  # –ï—Å–ª–∏ —Ñ—Ä—É–∫—Ç —Å–ª–µ–≤–∞ –±–æ–ª—å—à–µ, —á–µ–º —Ñ—Ä—É–∫—Ç —Å–ø—Ä–∞–≤–∞
                fruits[j], fruits[j + 1] = fruits[j + 1], fruits[j]  # –ú–µ–Ω—è—é –º–µ—Å—Ç–∞–º–∏
    return fruits

def insertion_sort(fruits: List[Tuple[str, int]]) -> List[Tuple[str, int]]:
    """
    –°–æ—Ä—Ç–∏—Ä—É–µ—Ç —Å–ø–∏—Å–æ–∫ —Ñ—Ä—É–∫—Ç–æ–≤ –ø–æ —Ä–∞–∑–º–µ—Ä—É, –∏—Å–ø–æ–ª—å–∑—É—è –∞–ª–≥–æ—Ä–∏—Ç–º "—Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ –≤—Å—Ç–∞–≤–∫–∞–º–∏".

    :param fruits: –°–ø–∏—Å–æ–∫ –∫–æ—Ä—Ç–µ–∂–µ–π (—Ä–∞–∑–º–µ—Ä, –∏–¥–µ–Ω—Ç–∏—Ñ–∏–∫–∞—Ç–æ—Ä).
    :type fruits: List[Tuple[str, int]]
    :return: –û—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã–π —Å–ø–∏—Å–æ–∫ –∫–æ—Ä—Ç–µ–∂–µ–π.
    :rtype: List[Tuple[str, int]]

    .. code-block:: python

        fruits = [("üçâ", 1), ("üçé", 2), ("üçê", 3)]
        insertion_sort(fruits) # –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç [("üçé", 2), ("üçê", 3), ("üçâ", 1)]
    """
    for i in range(1, len(fruits)): # –ù–∞—á–∏–Ω–∞—é —Å–æ –≤—Ç–æ—Ä–æ–≥–æ —Ñ—Ä—É–∫—Ç–∞ (–ø–µ—Ä–≤—ã–π —Å—á–∏—Ç–∞–µ—Ç—Å—è –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã–º)
        key = fruits[i] # –ë–µ—Ä—É —Å–ª–µ–¥—É—é—â–∏–π —Ñ—Ä—É–∫—Ç
        j = i - 1 # –ò–Ω–¥–µ–∫—Å –ø—Ä–µ–¥—ã–¥—É—â–µ–≥–æ —Ñ—Ä—É–∫—Ç–∞
        while j >= 0 and compare_fruits(fruits[j], key) == 1: # –ò—â—É –ø–æ–∑–∏—Ü–∏—é –≤ –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω–æ–π —á–∞—Å—Ç–∏, –∫—É–¥–∞ –≤—Å—Ç–∞–≤–∏—Ç—å —Ñ—Ä—É–∫—Ç
            fruits[j + 1] = fruits[j] # –°–¥–≤–∏–≥–∞—é —Ñ—Ä—É–∫—Ç—ã, —á—Ç–æ –±—ã –æ—Å–≤–æ–±–æ–¥–∏—Ç—å –º–µ—Å—Ç–æ –¥–ª—è –Ω–æ–≤–æ–≥–æ
            j -= 1
        fruits[j + 1] = key # –í—Å—Ç–∞–≤–ª—è—é —Ñ—Ä—É–∫—Ç –Ω–∞ –Ω—É–∂–Ω–æ–µ –º–µ—Å—Ç–æ
    return fruits

def selection_sort(fruits: List[Tuple[str, int]]) -> List[Tuple[str, int]]:
    """
    –°–æ—Ä—Ç–∏—Ä—É–µ—Ç —Å–ø–∏—Å–æ–∫ —Ñ—Ä—É–∫—Ç–æ–≤ –ø–æ —Ä–∞–∑–º–µ—Ä—É, –∏—Å–ø–æ–ª—å–∑—É—è –∞–ª–≥–æ—Ä–∏—Ç–º "—Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ –≤—ã–±–æ—Ä–æ–º".

    :param fruits: –°–ø–∏—Å–æ–∫ –∫–æ—Ä—Ç–µ–∂–µ–π (—Ä–∞–∑–º–µ—Ä, –∏–¥–µ–Ω—Ç–∏—Ñ–∏–∫–∞—Ç–æ—Ä).
    :type fruits: List[Tuple[str, int]]
    :return: –û—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã–π —Å–ø–∏—Å–æ–∫ –∫–æ—Ä—Ç–µ–∂–µ–π.
    :rtype: List[Tuple[str, int]]

    .. code-block:: python

        fruits = [("üçâ", 1), ("üçé", 2), ("üçê", 3)]
        selection_sort(fruits) # –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç [("üçé", 2), ("üçê", 3), ("üçâ", 1)]
    """
    n = len(fruits) # –ü–æ–ª—É—á–∞—é –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ñ—Ä—É–∫—Ç–æ–≤ –≤ —Å–ø–∏—Å–∫–µ
    for i in range(n): # –ü—Ä–æ—Ö–æ–∂—É –ø–æ –≤—Å–µ–º —Ñ—Ä—É–∫—Ç–∞–º –≤ —Å–ø–∏—Å–∫–µ
        min_index = i # –ò–Ω–¥–µ–∫—Å —Å–∞–º–æ–≥–æ –º–∞–ª–µ–Ω—å–∫–æ–≥–æ —Ñ—Ä—É–∫—Ç–∞
        for j in range(i + 1, n): # –ò—â—É —Å–∞–º—ã–π –º–∞–ª–µ–Ω—å–∫–∏–π —Ñ—Ä—É–∫—Ç –≤ –Ω–µ–æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω–æ–π —á–∞—Å—Ç–∏
            if compare_fruits(fruits[j], fruits[min_index]) == -1: # –ï—Å–ª–∏ –Ω–∞—à–µ–ª —Ñ—Ä—É–∫—Ç –º–µ–Ω—å—à–µ —Ç–µ–∫—É—â–µ–≥–æ –º–∏–Ω–∏–º—É–º–∞
                min_index = j # –ó–∞–ø–æ–º–∏–Ω–∞—é –∏–Ω–¥–µ–∫—Å –Ω–æ–≤–æ–≥–æ –º–∏–Ω–∏–º—É–º–∞
        fruits[i], fruits[min_index] = fruits[min_index], fruits[i] # –ú–µ–Ω—è—é —Ç–µ–∫—É—â–∏–π —Ñ—Ä—É–∫—Ç —Å —Å–∞–º—ã–º –º–∞–ª–µ–Ω—å–∫–∏–º –∏–∑ –Ω–µ–æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω–æ–π —á–∞—Å—Ç–∏
    return fruits

    """

    ("üçâ", 1), ("üçé", 2), ("üçê", 3), ("üß∫", 4), ("üçé", 5), ("üçâ", 6), ("üçê", 7),

logger.info("–ò—Å—Ö–æ–¥–Ω—ã–π —Å–ø–∏—Å–æ–∫ —Ñ—Ä—É–∫—Ç–æ–≤: " + display_fruits(fruits))  # –í—ã–≤–æ–∂—É –∏—Å—Ö–æ–¥–Ω—ã–π —Å–ø–∏—Å–æ–∫
logger.info("–ü—Ä–∏–º–µ—Ä—ã: –Ø–±–ª–æ–∫–æ (üçé) < –ì—Ä—É—à–∏ (üçê) < –î—ã–Ω—è (üçâ) < –ö–æ—Ä–∑–∏–Ω—ã (üß∫)") # –ü–æ–∫–∞–∑—ã–≤–∞—é –ø–æ—Ä—è–¥–æ–∫ —Ñ—Ä—É–∫—Ç–æ–≤

logger.info("–°–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ –ø—É–∑—ã—Ä—å–∫–æ–º: " + display_fruits(sorted_fruits_bubble)) # –í—ã–≤–æ–∂—É —Ä–µ–∑—É–ª—å—Ç–∞—Ç

# –°–æ—Ä—Ç–∏—Ä—É—é –≤—Å—Ç–∞–≤–∫–∞–º–∏
